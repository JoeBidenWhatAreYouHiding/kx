is_working_with_JAP = {
	if = {
		limit = {
			FNG_JAP_hostile = yes
		}
		OR = {
			AND = {
				tag = SQI
				has_country_leader = {
					name = "Zhang Zongchang"
					ruling_only = yes
				}
			}
			AND = {
				tag = GXC
				has_government = market_liberal
				has_country_flag = GXC_JAP_alliance
			}
		}
	}
	else = {
		is_ruled_by_pro_fengtian = yes
	}
}

JAP_chinese_enemy_exists = {
	event_target:current_china_leader = {
		exists = yes
		NOT = {
			has_capitulated = yes
			is_subject_of = JAP
			AND = {
				is_subject = yes
				overlord = { is_working_with_JAP = yes }
			}
		}
	}
}
