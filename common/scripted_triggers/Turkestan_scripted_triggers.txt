TRK_equipment_decisions_target_check = {
	if = {
		limit = { 
			FROM = { tag = INT }
		}
		OR = {
			is_ally_with = INT
			if = {
				limit = { ROOT = { tag = KHI } }
				has_idea = KHI_communard_advisors
			}
			else = {
				has_socialist_government = yes
			}
		}
		NOT = {
			is_ally_with = ENT
			is_ally_with = TUR
			is_ally_with = RUS
			is_ally_with = GER
		}
	}
	else_if = {
		limit = { 
			FROM = { tag = ENT }
		}
		OR = {
			is_ally_with = ENT
			has_completed_focus = UZB_British_Specialists
		}
		NOT = {
			is_ally_with = INT
			is_ally_with = TUR
			is_ally_with = RUS
			is_ally_with = GER
		}
	}
	else_if = {
		limit = { 
			FROM = { tag = TUR }
		}
		OR = {
			is_ally_with = TUR
			if = {
				limit = { tag = KHI }
				has_idea = KHI_ottoman_advisors
			}
		}
		NOT = {
			has_socialist_government = yes
			is_ally_with = ENT
			is_ally_with = INT
			is_ally_with = RUS
			is_ally_with = GER
		}
	}
	else_if = {
		limit = { 
			FROM = { tag = RUS }
		}
		OR = {
			is_ally_with = RUS
			if = {
				limit = { tag = UZB }
				has_country_flag = UZB_russian_specialists
				UZB_has_russian_hegemony = yes
			}
			else_if = {
				limit = { tag = TRK }
				has_completed_focus = TRK_Russian_Aid
			}
			else_if = {
				limit = { tag = ALO }
				ALO_has_russian_influence = yes
			}
			else_if = {
				limit = { tag = KYR }
				ALO = { ALO_has_russian_influence = yes }
			}
		}
		NOT = {
			is_ally_with = ENT
			is_ally_with = TUR
			is_ally_with = INT
			is_ally_with = GER
		}
	}
	else_if = {
		limit = { 
			FROM = { tag = GER }
		}
		if = {
			limit = { is_ally_with = GER }
			is_ally_with = GER
		}
		else = {
			NOT = {
				has_socialist_government = yes
				has_idea = KHI_communard_advisors
				has_completed_focus = UZB_British_Specialists
				has_country_flag = UZB_russian_specialists
				UZB_has_russian_hegemony = yes
				tag = ALO
				tag = KYR
				is_ally_with = RUS
				has_completed_focus = TRK_Russian_Aid
			}
		}
		NOT = {
			is_ally_with = ENT
			is_ally_with = INT
			is_ally_with = RUS
			is_ally_with = TUR
		}
	}
}
