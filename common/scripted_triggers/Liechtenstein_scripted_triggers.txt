LIC_im_staying_in_my_lane = { #Does my customs union align with my position on the BoP?
	if = {
		limit = {
			OR = {
				NOT = {
					has_completed_focus = LIC_continue_customs_union
					has_completed_focus = LIC_foreign_suspicion
					has_completed_focus = LIC_Swiss_Compact
				}
				has_completed_focus = LIC_continue_customs_union
			}
		}
		power_balance_value = {
			id = LIC_austro_swiss
			value > 0.1
		}
	}
	if = {
		limit = {
			OR = {
				has_completed_focus = LIC_Swiss_Compact
			}
		}
		power_balance_value = {
			id = LIC_austro_swiss
			value < -0.1
		}
	}
	if = {
		limit = {
			OR = {
				has_completed_focus = LIC_foreign_suspicion
			}
		}
		AND = {
			power_balance_value = {
				id = LIC_austro_swiss
				value < -0.1
			}
			power_balance_value = {
				id = LIC_austro_swiss
				value > 0.1
			}
		}
	}
}
