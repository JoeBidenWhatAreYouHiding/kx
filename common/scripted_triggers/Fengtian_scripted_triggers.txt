FNG_can_attend_unification_conference = {
	is_subject = no
	is_in_faction = no
	NOT = { has_country_flag = united_front_member }
	OR = {
		is_ruled_by_pro_fengtian = yes
		AND = {
			tag = SZC
			OR = {
				has_government = market_liberal
				has_government = social_conservative
				has_government = paternal_autocrat
			}
		}
		AND = {
			tag = XSM
			NOT = { has_government = social_conservative }
		}
		AND = {
			tag = SHX
			has_government = paternal_autocrat
		}
		AND = {
			tag = SIK
			has_country_leader = {
				name = "Ma Shaowu"
				ruling_only = yes
			}
		}
		AND = {
			tag = TAN
			OR = {
				has_country_leader = {
					name = "Ma Zhongying"
					ruling_only = yes
				}
				has_country_leader = {
					name = "Hoja-Niyaz"
					ruling_only = yes
				}
			}
		}
		AND = {
			tag = YUN
			has_country_leader = {
				name = "Long Yun"
				ruling_only = yes
			}
		}
	}
}
