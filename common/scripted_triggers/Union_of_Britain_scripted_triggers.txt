#Military factories
ENG_FED_military_factory_stronger = {
	set_temp_variable = { ENG_FED_conference_milfactory_target = FROM.num_of_military_factories }
	set_temp_variable = { ENG_FED_conference_milfactory = ENG.num_of_military_factories }
	divide_temp_variable = { 
		ENG_FED_conference_milfactory = ENG_FED_conference_milfactory_target
	}
	check_variable = {
		ENG_FED_conference_milfactory > 1.33
	}
}
ENG_FED_military_factory_equal = {
	set_temp_variable = { ENG_FED_conference_milfactory_target = FROM.num_of_military_factories }
	set_temp_variable = { ENG_FED_conference_milfactory = ENG.num_of_military_factories }
	divide_temp_variable = { 
		ENG_FED_conference_milfactory = ENG_FED_conference_milfactory_target
	}
	check_variable = {
		ENG_FED_conference_milfactory > 0.75
	}
}
ENG_FED_military_factory_weaker = {
	set_temp_variable = { ENG_FED_conference_milfactory_target = FROM.num_of_military_factories }
	set_temp_variable = { ENG_FED_conference_milfactory = ENG.num_of_military_factories }
	divide_temp_variable = { 
		ENG_FED_conference_milfactory = ENG_FED_conference_milfactory_target
	}
	check_variable = {
		compare = less_than_or_equals
		var = ENG_FED_conference_milfactory
		value = 0.75
	}
}
