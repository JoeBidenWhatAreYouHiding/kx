is_country_in_faction_hostile_to_germany = {
	OR = {
		is_in_faction_with = INT
		is_in_faction_with = SER
		AND = {
			is_in_faction = yes
			faction_leader = {
				OR = {
					original_tag = JAP
					is_russian_tag_that_can_integrate = yes
				}
				is_subject = no
			}
		}
	}
}
