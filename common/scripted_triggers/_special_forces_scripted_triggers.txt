can_unlock_new_special_forces_tree = {
	if = {
		limit = { NOT = { has_variable = sf_trees } }
		custom_override_tooltip = {
			tooltip = can_unlock_new_special_forces_tree_none
			NOT = { has_variable = sf_trees }
		}
	}
	else_if = {
		limit = { check_variable = { sf_trees > 0 } }
		check_variable = {
			tooltip = can_unlock_new_special_forces_tree_more_yes
			sf_trees > 0
		}
	}
	else = {
		custom_override_tooltip = {
			tooltip = can_unlock_new_special_forces_tree_more_no
			always = no
		}
	}
}
can_unlock_second_track_of_sf_doctrine = {
	if = {
		limit = { check_variable = { sf_trees > 1 } }
		custom_trigger_tooltip = {
			tooltip = can_unlock_second_track_special_forces_doctrine_yes
			check_variable = { sf_trees > 1 }
		}
	}
	else = {
		custom_trigger_tooltip = {
			tooltip = can_unlock_second_track_special_forces_doctrine_no
			always = no
		}
	}
}
