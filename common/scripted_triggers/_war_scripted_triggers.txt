russian_civil_war = {
	custom_trigger_tooltip = {
		tooltip = RUS_cw_ongoing_tt
		OR = {
			RUS = {
				OR = {
					has_war_with = RVA 
					has_war_with = RRF 
					has_war_with = WRA 
					has_war_with = TRM 
				}
			}
			RVA = { has_war_with = RRF }
			WRA = { has_war_with = RRF }
			RRF = { has_war_with = WRA }
			RVA = { has_war_with = TRM }
			TRM = { has_war_with = WRA }
			RRF = { has_war_with = TRM }
			has_global_flag = RUS_cw_begun
		}
		NOT = { has_global_flag = RUS_cw_over }
	}
}

		has_war_with = ROOT

		ROOT = { is_justifying_wargoal_against = PREV }

	}

minor_faction_member_cannot_start_inter_faction_war_before_wk = {
	if = {
		limit = {
			is_in_faction = yes
			is_faction_leader = no
			faction_leader = { is_actual_major = yes }
			FROM = {
				OR = {
					AND = {
						is_in_faction = yes
						faction_leader = { is_actual_major = yes }
					}
					AND = {
						has_guarantor = yes
						any_of_scopes = {
							array = guarantor_array
							has_guaranteed = PREV
							is_actual_major = yes
							NOT = { has_war_with = ROOT }
							NOT = { has_war_with = ROOT.faction_leader }
						}
					}
				}
			}
		}
		second_weltkrieg_can_start = yes
	}
}
