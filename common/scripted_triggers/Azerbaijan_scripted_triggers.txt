AZE_radsoc_win = {

	set_temp_variable = { AZE_marlib = AZE.party_popularity_100@market_liberal }
	
	set_temp_variable = { AZE_radsoc = AZE.party_popularity_100@radical_socialist}
	set_temp_variable = { AZE_socdem = AZE.party_popularity_100@social_democrat}
	add_to_temp_variable = { AZE_socdem = AZE.party_popularity_100@totalist}
	set_temp_variable = { AZE_soclib = AZE.party_popularity_100@social_liberal}
	set_temp_variable = { AZE_soccon = AZE.party_popularity_100@social_conservative}
	set_temp_variable = { AZE_authdem = AZE.party_popularity_100@authoritarian_democrat}
	add_to_temp_variable = { AZE_authdem = AZE.party_popularity_100@paternal_autocrat}
	
	check_variable = { AZE_radsoc > AZE_socdem }
	check_variable = { AZE_radsoc > AZE_soclib }
	check_variable = { AZE_radsoc > AZE_soccon }
	check_variable = { AZE_radsoc > AZE_authdem }
	
	set_temp_variable = { sum = AZE_marlib }
	add_to_temp_variable = { sum = AZE_radsoc }
	add_to_temp_variable = { sum = AZE_socdem }
	add_to_temp_variable = { sum = AZE_soclib }
	add_to_temp_variable = { sum = AZE_soccon }
	add_to_temp_variable = { sum = AZE_authdem }
	
	divide_temp_variable = { AZE_radsoc = sum }
	
	
}
