SPR_bop_countdown_adjust_effect = {
	set_temp_variable = { temp_mission_duration = days_mission_timeout@SPR_bop_opposing_side_does_something_mission }
	log = "days_mission_timeout: [?days_mission_timeout@SPR_bop_opposing_side_does_something_mission]"
	log = "temp_mission_duration:[?temp_mission_duration]"
	log = "radicals_mission_duration_limit: [?SPR.radicals_mission_duration_limit]"
	if = {
		limit = {
			check_variable = { temp_mission_duration > SPR.radicals_mission_duration_limit }
		}
		subtract_from_temp_variable = { temp_mission_duration = SPR.radicals_mission_duration_limit }
		multiply_temp_variable = { temp_mission_duration = -1 }
		add_days_mission_timeout = {
			mission = SPR_bop_opposing_side_does_something_mission
			days = temp_mission_duration
		}
	}
}
