ZGS_recalculate_levies = {
    set_variable = { ZGS_active_levies = 0 }
    if = {
        limit = { has_country_flag = ZGS_qashqai_active }
        add_to_variable = { ZGS_active_levies = 1 }
    }
    if = {
        limit = { has_country_flag = ZGS_bakhtiari_active }
        add_to_variable = { ZGS_active_levies = 1 }
    }
    if = {
        limit = { has_country_flag = ZGS_luri_active }
        add_to_variable = { ZGS_active_levies = 1 }
    }
    if = {
        limit = { has_country_flag = ZGS_khamseh_active }
        add_to_variable = { ZGS_active_levies = 1 }
    }
    if = {
        limit = { has_country_flag = ZGS_mamasani_active }
        add_to_variable = { ZGS_active_levies = 1 }
    }
    if = {
        limit = { has_country_flag = ZGS_khuzestani_active }
        add_to_variable = { ZGS_active_levies = 1 }
    }
    if = {
    limit = {
        has_country_flag = ZGS_overextended
        NOT = { has_idea = ZGS_tribal_strain }
        NOT = { has_idea = ZGS_tribal_overstretch }
        NOT = { has_idea = ZGS_tribal_crisis }
    }
    country_event = { id = zagros.40 days = 7 }
    }
    if = {
        limit = {
            check_variable = { ZGS_active_levies > var:ZGS_levy_cap }
        }
        set_country_flag = ZGS_overextended
    }
    else = {
        clr_country_flag = ZGS_overextended
    }
}

ZGS_reduce_levies_to_cap = {
    if = {
        limit = {
            check_variable = { ZGS_active_levies > var:ZGS_levy_cap }
            has_country_flag = ZGS_khuzestani_active
        }
        clr_country_flag = ZGS_khuzestani_active
        remove_ideas = ZGS_tribe_khuzestani_active
        add_to_variable = { ZGS_active_levies = -1 }
    }
    if = {
        limit = {
            check_variable = { ZGS_active_levies > var:ZGS_levy_cap }
            has_country_flag = ZGS_khamseh_active
        }
        clr_country_flag = ZGS_khamseh_active
        remove_ideas = ZGS_tribe_khamseh_active
        add_to_variable = { ZGS_active_levies = -1 }
    }
    if = {
        limit = {
            check_variable = { ZGS_active_levies > var:ZGS_levy_cap }
            has_country_flag = ZGS_mamasani_active
        }
        clr_country_flag = ZGS_mamasani_active
        remove_ideas = ZGS_tribe_mamasani_active
        add_to_variable = { ZGS_active_levies = -1 }
    }
    if = {
        limit = {
            check_variable = { ZGS_active_levies > var:ZGS_levy_cap }
            has_country_flag = ZGS_luri_active
        }
        clr_country_flag = ZGS_luri_active
        remove_ideas = ZGS_tribe_luri_active
        add_to_variable = { ZGS_active_levies = -1 }
    }
    if = {
        limit = {
            check_variable = { ZGS_active_levies > var:ZGS_levy_cap }
            has_country_flag = ZGS_bakhtiari_active
        }
        clr_country_flag = ZGS_bakhtiari_active
        remove_ideas = ZGS_tribe_bakhtiari_active
        add_to_variable = { ZGS_active_levies = -1 }
    }
    if = {
        limit = {
            check_variable = { ZGS_active_levies > var:ZGS_levy_cap }
            has_country_flag = ZGS_qashqai_active
        }
        clr_country_flag = ZGS_qashqai_active
        remove_ideas = ZGS_tribe_qashqai_active
        add_to_variable = { ZGS_active_levies = -1 }
    }
    add_to_variable = { ZGS_tribal_dirty = 1 }
}
