FIU_coup_raise = {
	add_to_variable = {
		var = FIU_coup_value
		value = 1
	}
	if = {
		limit = {
			check_variable = {
				var = FIU_coup_value
				value = 5
				compare = less_than
			}
			has_country_flag = FIU_coups_over
		}
		random_list = {
			20 = {
				country_event = {
					id = nufiucoup.1
					days = 15
					random = 200
				}
			}
			20 = {
				country_event = {
					id = nufiucoup.2
					days = 15
					random = 200
				}
			}
			20 = {
				country_event = {
					id = nufiucoup.3
					days = 15
					random = 200
				}
			}
			20 = {
				country_event = {
					id = nufiucoup.4
					days = 15
					random = 200
				}
			}
			20 = {
				country_event = {
					id = nufiucoup.5
					days = 15
					random = 200
				}
			}
		}
		else = {
			country_event = {
				id = nufiucoup.6
				days = 15
				random = 200
			}
		}
	}
}
