    add_to_variable = {
        var = THIS.development_value
        value = 1
    }
    CEN_raise_development_check = yes
}

CEN_raise_development_2 = {
    add_to_variable = { THIS.development_value = 2 }
    CEN_raise_development_check = yes
}

CEN_lower_development_2 = {
	subtract_from_variable = { THIS.development_value = 2 }
	CEN_raise_development_check = yes
}

    if = {
        limit = {
            check_variable = { THIS.development_value > 3 }
        }
        if = {
            limit = { has_idea = CEN_development_idea_1 }
            swap_ideas = {
                remove_idea = CEN_development_idea_1
                add_idea = CEN_development_idea_2
            }
        }
    }
    else_if = {
        limit = {
            check_variable = { THIS.development_value < 4 }
        }
        if = {
            limit = { has_idea = CEN_development_idea_2 }
            swap_ideas = {
                remove_idea = CEN_development_idea_2
                add_idea = CEN_development_idea_1
            }
        }
    }
}
