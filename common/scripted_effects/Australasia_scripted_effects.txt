	every_other_country = {
		limit = {
			OR = {
				has_socialist_government = yes
				is_in_faction_with = INT
			}
		}
		clear_embargo_PREV = yes
	}
	every_other_country = {
		limit = { is_in_faction_with = ENT }
		embargo_PREV = yes
	}
