LIC_vibe_check_famine = {
	#Check if readiness is below 0; apply famine effect!
	if = {
		limit = {
			check_variable = {
				var = LIC.govt_readiness
				value = 0
				compare = less_than
			}
			
		}
		add_to_variable = { var = LIC.famine_tally_var value = 1 } #If it reaches 12, let the VDBL coup!
		LIC_famine_tally = yes
		if = {
			limit = {
				NOT = {
					has_dynamic_modifier = {
						modifier = LIC_famine
					}
				}
			}
			custom_effect_tooltip = LIC_famine_start
			set_variable = { var = LIC.famine_value value = LIC.govt_readiness }
			add_dynamic_modifier = {
				modifier = LIC_famine
			}
			LIC_famine_tally = yes
		}
		
	}
	if = {
		limit = {
			has_dynamic_modifier = {
				modifier = LIC_famine
			}
			check_variable = {
				var = LIC.govt_readiness
				value = 0
				compare = greater_than
			}
			check_variable = {
				var = LIC.famine_value
				value = 0
				compare = greater_than_or_equals
			}
		}
		custom_effect_tooltip = LIC_famine_end
		remove_dynamic_modifier = {
			modifier = LIC_famine
		}
	}
}

				compare = equals
			}
		}
