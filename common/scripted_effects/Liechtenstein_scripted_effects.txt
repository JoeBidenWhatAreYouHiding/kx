LIC_vibe_check_famine = {
	#Check if readiness is below 0; apply famine effect!
	if = {
		limit = {
			check_variable = {
				var = LIC.govt_readiness
				value = 0
				compare = less_than
			}
			NOT = {
				has_country_flag = LIC_vibecheck
			}
		}
		if = {
			limit = {
				NOT = {
					has_country_flag = LIC_quotas_done
				}
			}
			add_to_variable = { LIC.famine_tally_var = 1 } #If it reaches 24, let the VDBL coup!
			if = {
				limit = {
					check_variable = { LIC.famine_tally_var = 8 }
				}
				country_event = licflavor.73
			}
			if = {
				limit = {
					check_variable = { LIC.famine_tally_var = 12 }
				}
				country_event = licflavor.77
			}
			if = {
				limit = {
					check_variable = { LIC.famine_tally_var = 16 }
				}
				country_event = licflavor.78
			}
			if = {
				limit = {
					check_variable = { LIC.famine_tally_var = 20 }
				}
				country_event = licflavor.79
			}
			if = {
				limit = {
					check_variable = {
						var = LIC.famine_tally_var
						value = 24
						compare = greater_than_or_equals
					}
				}
				country_event = lic.58
			}
			if = {
				limit = {
					NOT = { has_dynamic_modifier = { modifier = LIC_famine } }
				}
				custom_effect_tooltip = LIC_famine_start
				set_variable = { LIC.famine_value = 0 }
				add_dynamic_modifier = { modifier = LIC_famine }
			}
			LIC_famine_tally = yes
			else_if = {
				limit = {
					NOT = {
						has_country_flag = LIC_empty_stockpile_flag
					}
				}
				set_country_flag = LIC_empty_stockpile_flag
				add_idea = LIC_empty_stockpile
			}
		}
	}
	if = {
		limit = {
			NOT = {
				has_country_flag = LIC_vibecheck
			}
		}
		LIC_famine_addition = yes
	}
	if = {
		limit = {
			has_dynamic_modifier = { modifier = LIC_famine }
			check_variable = {
				var = LIC.famine_value
				value = 0
				compare = greater_than_or_equals
			}
		}
		custom_effect_tooltip = LIC_famine_end
		remove_dynamic_modifier = { modifier = LIC_famine }
	}
}
