BAS_add_soccon_hog = {
	if = {
		limit = { BAS_has_monarchy = yes }
		if = { 
			limit = { has_country_flag = BAS_irujo }
			add_ideas = BAS_Manuel_de_Irujo_y_Ollo_hog_sco
		}
		else_if = { 
			limit = { has_country_flag = BAS_candido }
			add_ideas = BAS_candido_saseta_hog_sco
		}
		else_if = { 
			limit = { has_country_flag = BAS_arizmendiarrieta }
			add_ideas = BAS_Jose_Maria_Arizmendiarrieta_hog_sco
		}
		else_if = { 
			limit = { NOT = { has_country_flag = BAS_aguirre_dööd }}
			add_ideas = BAS_Jose_Antonio_de_Aguirre_y_Lecube_hog_sco
		}
	}
	else = { add_ideas = BAS_Manuel_de_Irujo_y_Ollo_hog_sco }
}
