	hidden_effect = {
		remove_dynamic_modifier = { modifier = RUS_backward_industry_modifier }
		add_dynamic_modifier = { modifier = RUS_industrialized_nation_modifier }		
	}
	swap_ideas = {
		remove_idea = rus_industry_lack
		add_idea = rus_industrialised_nation
	}
}

recalculate_monthly_industrialisation_rate = {
	
	set_variable = { RUS_backward_industry_industrialisation_rate = ROOT.num_of_factories }
	subtract_from_variable = { RUS_backward_industry_industrialisation_rate = RUS_prev_month_num_of_factories }
	divide_variable = { RUS_backward_industry_industrialisation_rate = ROOT.num_of_factories }
	
	set_variable = { RUS_prev_month_num_of_factories = ROOT.num_of_factories }
	
	if = {
		limit = {
			NOT = { has_idea = rus_industry_lack }
		}
		set_variable = { ROOT.RUS_backward_industry_efficiency_gain = 0.05 }
		set_variable = { ROOT.RUS_backward_industry_max_efficiency = 0.05 }
		set_variable = { ROOT.RUS_backward_industry_industrial_capacity = 0.1 }
		set_variable = { ROOT.RUS_backward_industry_production_speed_buildings = 0.05 }
		set_variable = { ROOT.RUS_backward_industry_research_speed = 0.03 }
		set_variable = { ROOT.RUS_backward_industry_political_power_gain = 0.05 }
		
		add_to_variable = { ROOT.RUS_backward_industry_efficiency_gain = RUS_backward_industry_industrialisation_rate }
		add_to_variable = { ROOT.RUS_backward_industry_max_efficiency = RUS_backward_industry_industrialisation_rate }
		add_to_variable = { ROOT.RUS_backward_industry_industrial_capacity = RUS_backward_industry_industrialisation_rate }
		add_to_variable = { ROOT.RUS_backward_industry_production_speed_buildings = RUS_backward_industry_industrialisation_rate }
		add_to_variable = { ROOT.RUS_backward_industry_research_speed = RUS_backward_industry_industrialisation_rate }
		add_to_variable = { ROOT.RUS_backward_industry_political_power_gain = RUS_backward_industry_industrialisation_rate }
	}
		force_update_dynamic_modifier = yes
}

recalculate_backward_industry_debuff = {
	if = {
		limit = {
			num_of_factories < 220
		}
		set_variable = { RUS_industrialisation_var = ROOT.num_of_factories }
		clamp_variable = {
			var = RUS_industrialisation_var
			min = 50
			max = 220
		}
		divide_variable = { RUS_industrialisation_var = 220 }
		clamp_variable = {
			var = RUS_industrialisation_var
			min = 0
			max = 1
		}
		set_variable = { RUS_industrialisation_2_var = RUS_industrialisation_var }
		set_variable = { RUS_industrialisation_3_var = RUS_industrialisation_var }
		set_variable = { RUS_industrialisation_4_var = RUS_industrialisation_var }
		
		set_temp_variable = { RUS_industrialisation_negative_factor = -1 }
		set_temp_variable = { RUS_industrialisation_negative_3_factor = -0.6 }
		set_temp_variable = { RUS_industrialisation_negative_4_factor = -0.85 }
		
		add_to_variable = { RUS_industrialisation_var = RUS_industrialisation_negative_factor }
		add_to_variable = { RUS_industrialisation_2_var = RUS_industrialisation_negative_factor }
		add_to_variable = { RUS_industrialisation_3_var = RUS_industrialisation_negative_3_factor }
		add_to_variable = { RUS_industrialisation_4_var = RUS_industrialisation_negative_4_factor }
		
		set_temp_variable = { RUS_industrialisation_negative_var = -0.1 }
		multiply_variable = { RUS_industrialisation_var = RUS_industrialisation_negative_var }
		set_temp_variable = { RUS_industrialisation_negative_2_var = -0.14 }
		multiply_variable = { RUS_industrialisation_2_var = RUS_industrialisation_negative_2_var }
		set_temp_variable = { RUS_industrialisation_negative_3_var = -0.08 }
		multiply_variable = { RUS_industrialisation_3_var = RUS_industrialisation_negative_3_var }
		set_temp_variable = { RUS_industrialisation_negative_4_var = -0.8 }
		multiply_variable = { RUS_industrialisation_4_var = RUS_industrialisation_negative_4_var }
		
		set_variable = { ROOT.RUS_backward_industry_efficiency_gain = -0.7 }
		set_variable = { ROOT.RUS_backward_industry_max_efficiency = -0.7 }
		set_variable = { ROOT.RUS_backward_industry_industrial_capacity = -1 }
		set_variable = { ROOT.RUS_backward_industry_production_speed_buildings = -1 }
		set_variable = { ROOT.RUS_backward_industry_research_speed = -1.9 }
		set_variable = { ROOT.RUS_backward_industry_political_power_gain = -1.3 }
	
		multiply_variable = { ROOT.RUS_backward_industry_efficiency_gain = RUS_industrialisation_var }
		multiply_variable = { ROOT.RUS_backward_industry_max_efficiency = RUS_industrialisation_var }
		multiply_variable = { ROOT.RUS_backward_industry_industrial_capacity = RUS_industrialisation_2_var }
		multiply_variable = { ROOT.RUS_backward_industry_production_speed_buildings = RUS_industrialisation_2_var }
		multiply_variable = { ROOT.RUS_backward_industry_research_speed = RUS_industrialisation_3_var }
		multiply_variable = { ROOT.RUS_backward_industry_political_power_gain = RUS_industrialisation_4_var }
		
		clear_variable = RUS_industrialisation_var
		clear_variable = RUS_industrialisation_2_var
		clear_variable = RUS_industrialisation_3_var
		clear_variable = RUS_industrialisation_4_var
		force_update_dynamic_modifier = yes
	}
}

RCW_spawn_militia_divisions = {
	### calculate number of units to spawn
	set_temp_variable = { units_to_spawn = state_population_k }
	divide_temp_variable = { units_to_spawn = 500 }

	### failsafe so at least one unit is always spawned
	clamp_temp_variable = { var = units_to_spawn min = 1 max = 12 }

	### tooltip
	set_temp_variable = { total_units_to_spawn = units_to_spawn }
	custom_effect_tooltip = RCW_recruit_units

	### spawn normal militias
	set_temp_variable = { counter = 1 }
	if = {
		limit = {
			check_variable = { var = units_to_spawn value = 1 compare = greater_than_or_equals }
		}
		ROOT = { create_RCW_militia_template = yes }
		for_loop_effect = {
			start = 1
			end = units_to_spawn
			compare = less_than_or_equals

			meta_effect = {
				text = {
					create_unit = {
						division = "name = [DIVISION] division_template = [TEMPLATE] start_experience_factor = 0.0"
						allow_spawning_on_enemy_provs = yes
						owner = ROOT
						id = 1776
					}
				}
				DIVISION = "\\\"[?counter]aya [THIS.GetName] [GetRCWMilitiaName]\\\""
				TEMPLATE = "\\\"[GetRCWMilitiaTemplate]\\\""
			}
			add_to_temp_variable = { counter = 1 }
		}
	}
}
