	set_temp_variable = { RUS_wrangel_influence_temp = RUS.wrangel_influence }

	if = {
		limit = {
			check_variable = { RUS_denikin_influence_temp = RUS_wrangel_influence_temp }
		}
		if = {
			limit = { has_idea = RUS_rovs_wrangel }
			swap_ideas = {
				remove_idea = RUS_rovs_wrangel
				add_idea = RUS_rovs_balance
			}
		}
		else_if = {
			limit = { has_idea = RUS_rovs_denikin }
			swap_ideas = {
				remove_idea = RUS_rovs_denikin
				add_idea = RUS_rovs_balance
			}
		}
	}
	else_if = {
		limit = {
			check_variable = { RUS_denikin_influence_temp > RUS_wrangel_influence_temp }
		}
		if = {
			limit = { has_idea = RUS_rovs_wrangel }
			swap_ideas = {
				remove_idea = RUS_rovs_wrangel
				add_idea = RUS_rovs_denikin
			}
		}
		else_if = {
			limit = { has_idea = RUS_rovs_balance }
			swap_ideas = {
				remove_idea = RUS_rovs_balance
				add_idea = RUS_rovs_denikin
			}
		}
	}
	else_if = {
		limit = {
			check_variable = { RUS_denikin_influence_temp < RUS_wrangel_influence_temp }
		}
		if = {
			limit = { has_idea = RUS_rovs_balance }
			swap_ideas = {
				remove_idea = RUS_rovs_balance
				add_idea = RUS_rovs_wrangel
			}
