enable_tech_cossack_corps = {
	custom_override_tooltip = {
		tooltip = {
			localization_key = tooltip_enable_hidden_tech
			TECH = country_culture|cossack_corps_tech
			EFFECT = tech_effect|cossack_corps_tech
		}
		set_technology = { cossack_corps_tech = 1 popup = no }
	}
}
disable_tech_cossack_corps = {
	set_technology = { cossack_corps_tech = 0 popup = no }
}
