AZE_remove_german_dependence = {
	GER_leave_mitteleuropa = yes
	if = {
		limit = { is_in_tech_sharing_group = german_research }
		remove_from_tech_sharing_group = german_research
	}
	if = {
		limit = { has_idea = AZE_Dependance_On_Germany }
		remove_ideas = AZE_Dependance_On_Germany
	}
	if = {
		limit = { has_idea = AZE_Dependance_On_Germany_1 }
		remove_ideas = AZE_Dependance_On_Germany_1
	}
	if = {
		limit = { has_idea = AZE_Dependance_On_Germany_2 }
		remove_ideas = AZE_Dependance_On_Germany_2
	}
	if = {
		limit = { has_idea = AZE_german_investments }
		remove_ideas = AZE_german_investments
	}
	GER = {
		diplomatic_relation = {
			country = ROOT
			relation = guarantee
			active = no
		}
		diplomatic_relation = {
			country = ROOT
			relation = military_access
			active = no
		}
	}
}
AZE_increase_armenian_insurgency = {
	if = {
		limit = { 
			NOT = { has_country_flag = AZE_end_armenian_insurgency_flag }
		}
		add_to_variable = { AZE_armenian_insurgency_var = 1 }
		clamp_variable = { var = AZE_armenian_insurgency_var min = 0 max = 3 }
		if = {
			limit = { has_idea = AZE_Armenian_Terrorism }
			swap_ideas = {
				remove_idea = AZE_Armenian_Terrorism
				add_idea = AZE_Armenian_Terrorism_1
			}
		}
		else_if = {
			limit = { has_idea = AZE_Armenian_Terrorism_1 }
			swap_ideas = {
				remove_idea = AZE_Armenian_Terrorism_1
				add_idea = AZE_Armenian_Terrorism_2
			}
		}
		else_if = {
			limit = { has_idea = AZE_Armenian_Terrorism_2 }
			swap_ideas = {
				remove_idea = AZE_Armenian_Terrorism_2
				add_idea = AZE_Armenian_Terrorism_3
			}
		}
		else_if = {
			limit = { has_idea = AZE_Armenian_Terrorism_3 }
			swap_ideas = {
				remove_idea = AZE_Armenian_Terrorism_3
				add_idea = AZE_Armenian_Terrorism_4
			}
		}
	}
}
AZE_decrease_armenian_insurgency = {
	if = {
		limit = { 
			NOT = { has_country_flag = AZE_end_armenian_insurgency_flag }
		}
		add_to_variable = { AZE_armenian_insurgency_var = -1 }
		clamp_variable = { var = AZE_armenian_insurgency_var min = 0 max = 3 }

			}
		}
		else_if = {
			limit = { has_idea = AZE_Armenian_Terrorism_2 }
			swap_ideas = {
				remove_idea = AZE_Armenian_Terrorism_2
				add_idea = AZE_Armenian_Terrorism_1
			}
		}
		else_if = {
			limit = { has_idea = AZE_Armenian_Terrorism_3 }
			swap_ideas = {
				remove_idea = AZE_Armenian_Terrorism_3
				add_idea = AZE_Armenian_Terrorism_2
			}
		}
		else_if = {
			limit = { has_idea = AZE_Armenian_Terrorism_4 }
			swap_ideas = {
				remove_idea = AZE_Armenian_Terrorism_4
				add_idea = AZE_Armenian_Terrorism_3
			}
		}
	}
}
AZE_end_armenian_insurgency = {
	set_country_flag = AZE_end_armenian_insurgency_flag
	remove_targeted_decision = { target = THIS decision = AZE_launch_raid_in_the_mountains }
	custom_effect_tooltip = tooltip_white_line
	if = {
		limit = { has_idea = AZE_Armenian_Terrorism }
		swap_ideas = {
			remove_idea = AZE_Armenian_Terrorism
			add_idea = AZE_remnants_of_armenian_insurgents_1
			days = 1460
		}
	}
	else_if = {
