AZE_akinchi_increase = {
	if = {
		limit = { has_elections = yes }
		if = {
			limit = { has_country_flag = AZE_akinchi_legalised }
			custom_effect_tooltip = AZE_akinchi_increase
		}
	}
	add_to_variable = { AZE_akinchi_strength = 10 }
	hidden_effect = {
		add_popularity = {
			ideology = syndicalist
			popularity = 0.02
		}
	}
}
