botde_ausgleich_countries = {
	input = game:all_countries
	operators = {
		limit = {
			has_country_flag = going_to_le_ausgleich
		}
	}
	name = BOTDE_COLLECTION_AUSGLEICH_COUNTRIES
}
botde_ausgleich_countries_except_austria = {
	input = collection:botde_ausgleich_countries
	operators = {
		limit = { NOT = { tag = AUS } }
	}
	name = BOTDE_COLLECTION_AUSGLEICH_COUNTRIES_EXCEPT_AUSTRIA
}
botde_ausgleich_countries_except_hungary = {
	input = collection:botde_ausgleich_countries
	operators = {
		limit = { NOT = { tag = HUN } }
	}
	name = BOTDE_COLLECTION_AUSGLEICH_COUNTRIES_EXCEPT_HUNGARY
}
botde_ausgleich_countries_except_austria_and_hungary = {
	input = collection:botde_ausgleich_countries
	operators = {
		limit = {
			NOT = {
				tag = AUS
				tag = HUN
			} 
		}
	}
	name = BOTDE_COLLECTION_AUSGLEICH_COUNTRIES_EXCEPT_AUSTRIA_AND_HUNGARY
}
