botde_constituents = {
	input = game:european_countries
	operators = {
		limit = {
			is_in_array = { global.constituents_array = THIS }
		}
	}
	name = BOTDE_COLLECTION_CONSTITUENTS
}
botde_minorites_countries = {
	input = game:all_possible_countries
	operators = {
		limit = {
			has_country_flag = CONST_minor
			is_in_array = { global.minorities_array = THIS.id }
		}
	}
	name = BOTDE_COLLECTION_MINORITIES_COUNTRIES
}
botde_constituents_and_minorites_countries = {
	input = game:all_possible_countries
	operators = {
		limit = {
			OR = {
				is_in_array = { global.minorities_array = THIS.id }
				is_in_array = { global.constituents_array = THIS.id }
			}
		}
	}
	name = BOTDE_COLLECTION_CONSTITUENTS_AND_MINORITIES_COUNTRIES
}
botde_constituents_or_minorites_countries_under_70_revolt = {
	input = collection:botde_constituents_and_minorites_countries
	operators = {
		limit = {
			check_variable = { THIS.subject_revolt_progress < 0.70 }
		}
	}
	name = BOTDE_COLLECTION_CONSTITUENTS_OR_MINORITIES_COUNTRIES_UNDER_70_REVOLT
}
botde_constituents_or_minorites_countries_over_70_revolt = {
	input = collection:botde_constituents_and_minorites_countries
	operators = {
		limit = {
			check_variable = { THIS.subject_revolt_progress > 0.69 }
		}
	}
	name = BOTDE_COLLECTION_CONSTITUENTS_OR_MINORITIES_COUNTRIES_OVER_70_REVOLT
}
botde_ausgleich_countries = {
	input = game:all_countries
	operators = {
		limit = {
			has_country_flag = going_to_le_ausgleich
		}
	}
	name = BOTDE_COLLECTION_AUSGLEICH_COUNTRIES
}
botde_ausgleich_countries_except_austria = {
	input = collection:botde_ausgleich_countries
	operators = {
		limit = { NOT = { tag = AUS } }
	}
	name = BOTDE_COLLECTION_AUSGLEICH_COUNTRIES_EXCEPT_AUSTRIA
}
botde_ausgleich_countries_except_hungary = {
	input = collection:botde_ausgleich_countries
	operators = {
		limit = { NOT = { tag = HUN } }
	}
	name = BOTDE_COLLECTION_AUSGLEICH_COUNTRIES_EXCEPT_HUNGARY
}
botde_ausgleich_countries_except_austria_and_hungary = {
	input = collection:botde_ausgleich_countries
	operators = {
		limit = {
			NOT = {
				tag = AUS
				tag = HUN
			} 
		}
	}
	name = BOTDE_COLLECTION_AUSGLEICH_COUNTRIES_EXCEPT_AUSTRIA_AND_HUNGARY
}
