BOTDE_minorities_in_GUI_score = {
	targets = {
		target_array = global.minorities_array
		target_non_existing = yes
		target_trigger = {
			FROM = {
				exists = no
				AUS_is_constituent = no
				NOT = { has_country_flag = no_minority_revolt }
				capital_scope = {
					owner = {
						OR = {
							tag = AUS
							is_subject_of = AUS
							AND = {
								HUN = { is_subject_of = AUS }
								is_subject_of = HUN
							}
						}
					}
				}
				OR = {
					meta_trigger = {
						#debug = yes
						text = {
							has_global_flag = [TAG]_on_array
						}
						TAG = "[FROM.GetTag]"
					}
					NOT = { #tags which only appear when the TAG_on_array global flag is set for them
						tag = CZE
						tag = BAN
						tag = KRK
						tag = PGU
						tag = VNC
					}
				}
			}
		}
		score = {
			base = 0
			modifier = {
				AUS_can_collapse = no
				factor = 0
			}
			modifier = {
				AND = {
					has_variable = subject_revolt_progress
					AUS_can_collapse = yes
					NOT = { has_global_flag = AUS_minority_array_not_sorted }
				}
				set_temp_variable = { temp_add = subject_revolt_progress }
				round_variable = temp_add
				add = temp_add
			}
			modifier = {
				AND = {
					has_global_flag = AUS_minority_array_not_sorted
					has_variable = minority_gui_priority
				}
				add = minority_gui_priority
			}
		}
	}
}
