#Military/economy weaker/stronger
defined_text = { #as your (ENG's) army gets bigger, ratio goes down
	name = ENG_FED_conference_military_strength
	text = {
		trigger = {
			FROM = {
				strength_ratio = {
					tag = ENG
					ratio > 1
				}
			}
		}
		localization_key = ENG_FED_conference_military_weaker
	}
	text = {
		trigger = {
			FROM = {
				OR = {
					strength_ratio = {
						tag = ENG
						ratio < 1
					}
					strength_ratio = {
						tag = ENG
						ratio = 1
					}
				}
				strength_ratio = {
					tag = ENG
					ratio > 0.675
				}
			}
		}
		localization_key = ENG_FED_conference_military_equal
	}
	text = {
		trigger = {
			FROM = {
				strength_ratio = {
					tag = ENG
					ratio < 0.675 #if your army is ~1.5x as big or more
				}
			}
		}
		localization_key = ENG_FED_conference_military_stronger
	}
}
