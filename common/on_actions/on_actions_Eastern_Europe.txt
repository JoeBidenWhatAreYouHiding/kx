on_actions = {
	on_startup = {
		effect = {
			ROM = {
				give_resource_rights = { state = 46 receiver = GER }
			}
		}
	}

	#ROOT is winner #FROM gets annexed - For civil wars on_civil_war_end is also fired
	on_annex = {
		effect = {
			if = {
				limit = {
					ROOT = { tag = POL }
					FROM = { tag = GAL }
				}
				ROM = {
					country_event = { id = poland.117 days = 3 }
				}
			}
			if = {
				limit = {
					tag = BAT
					FROM = {
						OR = {
							tag = EST
							tag = LAT
						}
					}
				}
				country_event = {
					id = balticduchy.195
					days = 1
				}
			}
		}
	}

	# ROOT is capitulated country, FROM is winner
	on_capitulation = {
		effect = {
			### RUS-INT peace over UKR
			if = {
				limit = {
					tag = UKR
					FROM = { is_russian_tag = yes }
					is_in_faction_with = INT
				}
				INT = {
					country_event = { id = ukraine.91 days = 2 }
				}
			}
		}
	}

	on_government_change = {
		effect = {
			if = {
				limit = {
					tag = POL
					has_socialist_government = no
				}
				event_target:POL_Konstanty_Rokossowski = { set_nationality = FRA }
			}
			else_if = {
				limit = {
					tag = POL
					has_socialist_government = yes
				}
				event_target:POL_Konstanty_Rokossowski = { set_nationality = POL }
			}
		}
	}

	on_new_term_election = {
		effect = {
			### Baltic Duchy ###
			if = {
				limit = { tag = BAT }
				country_event = balticduchy.170
				country_event = balticduchy.171
			}
			### Lithuania ###
			if = {
				limit = { tag = LIT }
				country_event = lithuania.11 #1938
			}
			### Poland ###
			if = {
				limit = { tag = POL }
				country_event = poland.46
				country_event = poland.47
				country_event = plcommon.15
			}
		}
	}

	#ROOT is winner #FROM is loser
	on_peaceconference_ended = {
		effect = {
			if = {
				limit = {
					ROOT = { tag = POL }
					FROM = { tag = GAL }
				}
				ROM = {
					country_event = { id = poland.117 days = 3 }
				}
			}
			if = {
				limit = {
					ROOT = { tag = LAT }
					FROM = { tag = BAT }
				}
				ROOT = { country_event = balticduchy.196 }
			}
			if = {
				limit = {
					ROOT = { tag = EST }
					FROM = { tag = BAT }
				}
				ROOT = { country_event = balticduchy.197 }
			}
		}
	}

	#ROOT is new controller #FROM is old controller #FROM.FROM is state ID
	on_state_control_changed = {
		effect = {
			### POL-UKR War
			if = {
				limit = {
					OR = {
						tag = POL
						tag = UKR
					}
					has_war_with = FROM
					FROM = {
						OR = {
							tag = POL
							tag = UKR
						}
					}
				}
				UKR = { country_event = poland.110 }
			}
		}
	}

	on_war = {
		effect = {
			# Polish events
			if = {
				limit = { tag = POL }
				country_event = {
					id = poland.226 #War with Germany!
					days = 1
					random_days = 1
				}
			}
	}
	on_declare_war = {  
		effect = { 
			if = { # Socialist Don no call ally
				limit = { 
					ROOT = {
						tag = DKB
						has_socialist_government = yes
					}  
					FROM = { tag = RUS } 
				} 
				DKB = { add_ideas = DKB_no_allies_cw }
			}
		} 
	}
}
