on_actions = {
	on_startup = {
		effect = {
			# Peace Conferences
			if = {
				limit = {
					has_game_rule = {
						rule = peace_conference_options
						option = PLAYER_PRIMACY
					}
				}
				set_global_flag = KR_player_leads_conferences
			}
			# toolpack
			if = {
					limit = {
							NOT = {
								has_game_rule = {
								rule = state_transfer_tool_options
								option = ACTIVE
								}
							}
					}
				set_global_flag = toolpack_hidden
			}
			# Division Limits
			if = {
				limit = {
					has_game_rule = {
						rule = unit_limit_options
						option = AI_ONLY
					}
				}
				set_global_flag = KR_unit_limits_ai_only
			}
			else_if = {
				limit = {
					has_game_rule = {
						rule = unit_limit_options
						option = NONE
					}
				}
				set_global_flag = KR_no_unit_limits
			}
			# Demobilization
			if = {
				limit = {
					has_game_rule = {
						rule = demobilization_options

			}

				limit = {
					has_game_rule = {
						rule = demobilization_options
						option = NONE
					}
				}
				set_global_flag = KR_no_demob
			}
			# Faction Buttons
			if = {
				limit = {
					has_game_rule = {
						rule = faction_button_options
						option = BUTTON_PRESENT
					}
				}
				set_global_flag = KR_faction_buttons_present
			}
			# Guarantees
			if = {
				limit = {
					has_game_rule = {
						rule = allow_guarantees
						option = LIMITED
					}
				}
				set_global_flag = KR_limited_guarantees_allowed
			}
			else_if = {
				limit = {
					has_game_rule = {
						rule = allow_guarantees
						option = ALWAYS_FREE
					}
				}
				set_global_flag = KR_always_guarantees_allowed
			}
			else_if = {
				limit = {
					has_game_rule = {
						rule = allow_guarantees
						option = SAME_IDEOLOGY
					}
				}
				set_global_flag = KR_same_ideology_guarantees_allowed
			}
			else_if = {
				limit = {
					has_game_rule = {
						rule = allow_guarantees
						option = BLOCKED
					}
				}
				set_global_flag = KR_no_guarantees_allowed
			}
			else_if = {
				limit = {
					has_game_rule = {
						rule = allow_revoke_guarantees
						option = BLOCKED
					}
				}
				set_global_flag = KR_no_allow_revoke_guarantees
			}
			
			### Military Access
			else_if = {
				limit = {
					has_game_rule = {
						rule = allow_access
						option = FREE
					}
				}
				set_global_flag = KR_free_allow_access
			}
			else_if = {
				limit = {
					has_game_rule = {
						rule = allow_access
						option = SAME_IDEOLOGY
					}
				}
				set_global_flag = KR_same_ideology_allow_access
			}
			else_if = {
				limit = {
					has_game_rule = {
						rule = allow_access
						option = BLOCKED
					}
				}
				set_global_flag = KR_no_allow_access
			}
			
			### Lend lease
			if = {
				limit = {
					has_game_rule = {
						rule = allow_lend_lease
						option = FREE
					}
				}
				set_global_flag = KR_free_allow_lend_lease
			}
			else_if = {
				limit = {
					has_game_rule = {
						rule = allow_lend_lease
						option = LIMITED
					}
				}
				set_global_flag = KR_limited_allow_lend_lease
			}
			else_if = {
				limit = {
					has_game_rule = {
						rule = allow_lend_lease
						option = SAME_IDEOLOGY
					}
				}
				set_global_flag = KR_same_ideology_allow_lend_lease
			}
			else_if = {
				limit = {
					has_game_rule = {
						rule = allow_lend_lease
						option = SAME_FACTION
					}
				}
				set_global_flag = KR_same_faction_allow_lend_lease
			}
			else_if = {
				limit = {
					has_game_rule = {
						rule = allow_lend_lease
						option = BLOCKED
					}
				}
				set_global_flag = KR_no_allow_lend_lease
			}

			if = {

						rule = allow_licensing

					has_game_rule = {
						rule = allow_licensing
						option = SAME_FACTION
					}
				}
				set_global_flag = KR_same_faction_allow_licensing
			}
			else_if = {
				limit = {
					has_game_rule = {
						rule = allow_licensing
						option = BLOCKED
					}
				}
				set_global_flag = KR_no_allow_licensing
			}
			
			### Wargoals
			if = {
				limit = {
					has_game_rule = {
						rule = allow_wargoals
						option = ALWAYS_FREE
					}
				}
				set_global_flag = KR_always_free_allow_wargoals
			}
			else_if = {
				limit = {
					has_game_rule = {
						rule = allow_wargoals
						option = LIMITED
					}
				}
				set_global_flag = KR_limited_allow_wargoals
			}
			else_if = {
				limit = {
					has_game_rule = {
						rule = allow_wargoals
						option = FREE_25
					}
				}
				set_global_flag = KR_free_25_allow_wargoals
			}
			else_if = {
				limit = {
					has_game_rule = {
						rule = allow_wargoals
						option = FREE_50
					}
				}
				set_global_flag = KR_free_50_allow_wargoals
			}
			else_if = {
				limit = {
					has_game_rule = {
						rule = allow_wargoals
						option = FREE_75
					}
				}
				set_global_flag = KR_free_75_allow_wargoals
			}
			else_if = {
				limit = {
					has_game_rule = {
						rule = allow_wargoals
						option = FREE_100
					}
				}
				set_global_flag = KR_free_100_allow_wargoals

			else_if = {
				limit = {
					has_game_rule = {
						rule = allow_wargoals
						option = FOCUSES_ONLY

				}

			if = {
				limit = {
					has_game_rule = {
						rule = allow_volunteers

				}

			else_if = {
				limit = {
					has_game_rule = {
						rule = allow_volunteers
						option = LIMITED
					}
				}
				set_global_flag = KR_limited_allow_volunteers
			}
			else_if = {
				limit = {
					has_game_rule = {
						rule = allow_volunteers
						option = SAME_IDEOLOGY
					}
				}
				set_global_flag = KR_same_ideology_allow_volunteers
			}
			else_if = {
				limit = {
					has_game_rule = {
						rule = allow_volunteers
						option = BLOCKED
					}
				}
				set_global_flag = KR_no_allow_volunteers
			}
		}
	}
}
