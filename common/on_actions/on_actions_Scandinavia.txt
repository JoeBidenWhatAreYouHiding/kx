on_actions = {
	on_startup = {
		effect = {
			#Coalitions
			DEN = {
				add_soclib_to_coalition = yes
			}
			NOR = {
				add_soclib_to_coalition = yes
				add_soccon_to_coalition = yes
			}
			SWE = {
				add_marlib_to_coalition = yes
			}
			FIN = {
				add_soccon_to_coalition = yes
			}
		}
	}

	on_government_change = {
		effect = {
			if = {
				limit = {
					tag = SWE
					has_socialist_government = yes
				}
				remove_ideas = SWE_union_activity
			}
		}
	}
	on_new_term_election = {
		effect = {
			### Denmark ###
			if = {
				limit = { tag = DEN }
				country_event = { id = den.60 }
				country_event = { id = den.61 }
			}
			### Iceland ###
			if = {
				limit = { tag = ICE }
				country_event = { id = ice.259 }
				country_event = { id = ice.260 }
			}
			### Norway ###
			else_if = {
				limit = { tag = NOR }
				country_event = norway.14	#1936
				country_event = norway.30	#1939
				country_event = norway.33	#1942
			}
			### Scandinavia ###
			else_if = {
				limit = { tag = SCA }
				country_event = scandinavia.3
			}
			### Sweden ###
			else_if = {
				limit = { tag = SWE }
				country_event = { id = swe.8 }
				country_event = { id = sweelect.5 }
				country_event = { id = sweelect.15 }
				country_event = { id = swe.55 }
			}
			### Finland ###
			else_if = {
				limit = { tag = FIN }
				country_event = { id = finland.142 }
				country_event = { id = finland.140 }
			}
		}
	}

	#FROM is country getting invited.
	on_offer_join_faction = {
		effect = {
			if = {
				limit = { tag = NOR }
				SWE = {
					country_event = {
						id = swe.45
						days = 2
						random_days = 2
					}
				}
			}
		}
	}

	on_peace = {
		effect = {
			# Swedish events
			if = {
				limit = { tag = SWE }
				country_event = {
					id = swe.127
					days = 2
					random_days = 4
				}
			}
			# Demilitarization of Aland
			else_if = {
				limit = {
					tag = FIN
					owns_state = 145
					is_subject = no
					has_socialist_government = no
					SWE = {
						exists = yes
						is_subject = no
						has_socialist_government = no
					}
				}
				145 = {
					set_demilitarized_zone = yes
				}
			}
		}
	}
	#ROOT is new controller #FROM is old controller #FROM.FROM is state ID
	on_state_control_changed = {
		effect = {
			if = {
				limit = { FROM = { tag = SWE } }
				country_event = swe.120
			}
		}
	}
}
