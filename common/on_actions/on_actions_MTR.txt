on_actions = {
	on_annex = {
		effect = {
			if = {
				limit = {
					tag = DAG
					FROM = { tag = MTR }
				}
				every_state = {
					limit = {
						is_core_of = FROM
					}
					add_core_of = ROOT
				}
			}
		}
	}
	on_state_control_changed = {
		effect = {
			if = {
				limit = {
					OR = {
						tag = MTR
						tag = GEO
					}
					has_war_with = FROM
					FROM = {
						OR = {
							tag = MTR
							tag = GEO
						}
					}
				}
				GEO = { country_event = mtr.290 }
				MTR = { country_event = mtr.292 }
			}
		}
	}
	on_weekly = {
		effect = {
			if = {
				limit = {
					MTR = {
						check_variable = {
							var = MTR_total_support
							value = 0.2
							compare = less_than
						}
						NOT = { has_country_flag = MTR_risk_coup }
					}
				}
				set_country_flag = MTR_risk_coup
				MTR = { country_event = mtr.106 }
			}
		}
	}
}
