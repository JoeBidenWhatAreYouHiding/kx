	on_new_term_election = {
		effect = {
			### Russia ###
			if = {
				limit = { tag = AZE }
				country_event = aze.1
			}
			if = {
				limit = {
					tag = RUS
					has_country_flag = soviet_russia_restored
					NOT = { has_government = totalist }
					NOT = {
						has_country_leader = {
							name = "Aleksey Rykov"
							ruling_only = yes
						}
					}
				}
				country_event = russoc.40
			}
		}
	}
	on_weekly = {
		effect = {
			### Russia ###
			if = {
				limit = {
					tag = RUS
					OR = {
						has_country_flag = snap_elections
						has_country_flag = RUS_elections_duma_1941
					}
				}
				RUS = {
					every_owned_state = {
						RUS_calculate_polling_averages = yes
					}
				}
			}			
		}
	}
}
