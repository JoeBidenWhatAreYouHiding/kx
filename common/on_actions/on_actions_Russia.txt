on_actions = {
	on_startup = {
		effect = {
			#Russian integration arrays
			for_each_scope_loop = {
				array = RUS.owned_states
				add_to_array = { RUS.RUS_starting_states_array = THIS }
			}
			for_each_scope_loop = {
				array = ALO.owned_states
				add_to_array = { RUS.RUS_kazakhstan_region = THIS }
			}
			for_each_scope_loop = {
				array = TRK.owned_states
				add_to_array = { RUS.RUS_bukhara_region = THIS }
			}
			for_each_scope_loop = {
				array = UZB.owned_states
				add_to_array = { RUS.RUS_khiva_region = THIS }
			}
			for_each_scope_loop = {
				array = KYR.owned_states
				add_to_array = { RUS.RUS_kokand_region = THIS }
			}
			880 = { #Dvinsk
				add_dynamic_modifier = { modifier = BAT_Latgale_autonomous_region }
			}
		}
	}

	#ROOT is winner #FROM gets annexed - For civil wars on_civil_war_end is also fired
	on_annex = {
		effect = {
			if = {
				limit = {
					tag = TRK
					FROM = { tag = ALO }
				}
				country_event = {
					id = turkestan.9
					days = 30
					random = 720
				}
			}
			if = {
				limit = {
					AND = {
						OR = {
							tag = RUS
							tag = SOV
						}
						FROM = { tag = TRM }
					}
				}
				if = {
					limit = {
						NOT = { has_country_flag = SOV_restored_in_transamur }
					}
					country_event = {
						id = tram.400
						days = 2
					}
				}
			}
			if = {
				limit = {
					AND = {
						ROOT = {
							tag = BRY 
							NOT = { country_exists = TRM }
							NOT = { country_exists = SIB }
							has_global_flag = TRM_civil_war
							BRY = { 
								OR = {
									has_government = paternal_autocrat
									has_government = totalist
								}
								owns_state = 408
							}
						}
						FROM = { tag = TRM }
					}
				}
				ROOT = { country_event = tram.303 }
			}
			if = {
				limit = {
					AND = {
						tag = RUS
						FROM = { tag = SOV }
					}
				}
				every_owned_state = {
					limit = {
						is_claimed_by = SOV
					}
					remove_claim_by = SOV 
				}
				every_owned_state = {
					limit = {
						is_core_of = SOV
					}
					add_core_of = RUS
					remove_core_of = SOV
				}
			}
			if = {
				limit = {
					AND = {
						tag = SOV
						FROM = { tag = RUS }
					}
				}
				every_owned_state = {
					limit = {
						is_core_of = RUS
					}
					add_core_of = SOV
				}
				if = {
					limit = {
						has_cosmetic_tag = TRM_NRA
					}
					drop_cosmetic_tag = yes
				}
			}			
		}
	} 
	on_capitulation = {
		effect = {
			# Disable SOV civil war unit lock
			if = {
				limit = {
					ROOT = {
						tag = SOV
						has_idea = SOV_chaotic_revolution
					}
				}
				ROOT = {
					remove_ideas = SOV_chaotic_revolution
					country_lock_all_division_template = no
				}
			}
			# Disband the Russian Academy of Sciences
			if = {
				limit = {
					tag = RUS
					is_in_tech_sharing_group = russian_research
				}
				remove_from_tech_sharing_group = russian_research
				every_other_country = {
					limit = { is_in_tech_sharing_group = russian_research }
					remove_from_tech_sharing_group = russian_research
				}
			}
		}
	}
	
	
	# ROOT is capitulated country, FROM is winner
	on_capitulation = {
		effect = {
			#Russian-Bulgarian alliance
			if = {
				limit = {
					tag = BUL
					is_in_faction_with = RUS
					NOT = {
						any_of_scopes = {
							array = RUS.faction_members
							is_subject = no
						}
					}
				}
				RUS = {
					country_event = { id = bulgaria.99 months = 2 }
				}
			}

			# Disband the Russian Academy of Sciences
			if = {
				limit = {
					tag = RUS
					is_in_tech_sharing_group = russian_research
				}
				remove_from_tech_sharing_group = russian_research
				every_other_country = {
					limit = { is_in_tech_sharing_group = russian_research }
					remove_from_tech_sharing_group = russian_research
				}
			}
		}
	}

	on_declare_war = {
		effect = {
			# Great Patriotic War (event)
			if = {
				limit = {
					OR = {
						AND = {
							tag = RUS
							FROM = {
								OR = {
									tag = GER
									tag = FRA
								}
							}
						}
						AND = {
							OR = {
								tag = GER
								tag = FRA
							}
							FROM = { tag = RUS }
						}
					}
				}
				RUS = {
					country_event = {
						id = rusfor.51
						days = 3
					}
				}
			}
			if = {
				limit = {
					OR = {
						AND = {
							tag = TRM
							FROM = {
								OR = {
									tag = SOV
									tag = RUS
								}
							}
						}
						AND = {
							OR = {
								tag = RUS
								tag = SOV
							}
							FROM = { tag = TRM }
						}
					}
				}
				if = {
					limit = {
						TRM = { is_ai = no }
					}
					TRM = { add_ideas = TRM_Siberian_War_Buff }
					hidden_effect = {
						SOV = { add_ideas = RUS_Siberian_War_Debuff }
						RUS = { add_ideas = RUS_Siberian_War_Debuff }
					}
				}
				else = {
					TRM = { add_ideas = TRM_Siberian_War_Debuff }
					hidden_effect = {
						RUS = { add_ideas = RUS_Siberian_War_Buff }
						SOV = { add_ideas = RUS_Siberian_War_Buff }
					}
				}
				hidden_effect = { 
					RUS = { 
						country_event = { 
							id = russia.1920 
							days = 400
						}
					}
					SOV = { 
						country_event = { 
							id = russia.1920 
							days = 400
						}
					}
					TRM = { 
						country_event = { 
							id = russia.1921
							days = 400
						}
						country_event = { 
							id = russia.1922
							days = 400
						}
					}
				}
			}
		}
	}

	#ROOT is winner #FROM is loser
	on_peaceconference_ended = {
		effect = {
			if = {
				limit = {
					ROOT = { tag = SOV }
					FROM = { tag = RUS }
				}
				ROOT = {
					country_event = { id = russia.1949 days = 1 random = 30 }
				}
			}
			if = {
				limit = {
					ROOT = { tag = RUS }
					FROM = { tag = SOV }
				}
				ROOT = {
					country_event = { id = russia.1951 days = 1 random = 30 }
				}
			}
		}
	}
	
	#ROOT is new controller #FROM is old controller #FROM.FROM is state ID
	on_state_control_changed = {
		effect = {
			### RUS/SOV control array
			if = {
				limit = {
					OR = {
						tag = RUS
						tag = SOV
					}
					NOT = { is_in_array = { ROOT.enemy_controlled_states = FROM.FROM } }
					FROM.FROM = {
						owner = { has_war_with = ROOT }
						OR = {
							is_on_continent = europe
							is_on_continent = middle_east
						}
					}
				}
				set_country_flag = has_enemy_control_array
				add_to_array = { ROOT.enemy_controlled_states = FROM.FROM }
			}

			### A New 1812/1328 (event)
			if = {
				limit = {
					FROM = { tag = RUS }
					FROM.FROM = {
						OR = {
							state = 252
							state = 255
							state = 205
							state = 242
							state = 246
						}
					}
				}
				FROM = {
					country_event = {
						id = rusfor.51
						hours = 6
					}
				}
			}

			# Fall of Yekaterinburg (event)
			if = {
				limit = {
					tag = SOV
					FROM = { tag = RUS }
					FROM.FROM = { state = 653 } #Yekaterinburg
					ROOT = { controls_province = 6338 }
				}
				country_event = russia.1947
				FROM = { country_event = russia.1948 }
			}
			
			# TCW - Japanese-controlled Vladivostok(event)
			if = {
				limit = {
					OR = {
						tag = SIB
						tag = TRM
					}
					FROM = { 
						tag = JAP
					}
					FROM.FROM = { state = 408 } #Yekaterinburg
					ROOT = { controls_province = 957 }
					FROM = { NOT = { owns_state = 408 } }
				}
				custom_effect_tooltip = KR_separate_peace
				FROM = { white_peace = ROOT }
			}
			# TCW - Buryatia
			if = {
				limit = {
					ROOT = {
						tag = BRY
						NOT = { has_government = totalist }
						NOT = { has_government = paternal_autocrat }
					}
					FROM = { 
						tag = TRM
					}
					FROM.FROM = { state = 408 } #Yekaterinburg
					ROOT = { controls_province = 957 }
				}
				custom_effect_tooltip = KR_separate_peace
				FROM = { white_peace = ROOT }
				BRY = { transfer_state = 565 }
				if = {
					limit = {
						NOT = { country_exists = YAK }
					}
					YAK = {
						transfer_state = 574
						add_state_core = 574
						set_country_flag = YAK_secured_its_independence
						white_peace = TRM
					}
				}
				FROM = {
					add_war_support = -0.25
					add_stability = -0.25
					add_popularity = {
						ideology = authoritarian_democrat
						popularity = 0.15
					}
					remove_ideas = TRM_dream
					diplomatic_relation = {
						country = ROOT
						relation = non_aggression_pact
						active = yes
					}
					diplomatic_relation = {
						country = YAK
						relation = non_aggression_pact
						active = yes
					}						
				}
			}
			if = {
				limit = {
					tag = TRM
					FROM = { 
						OR = {
							tag = SOV
							tag = RUS
						}
					}
					FROM.FROM = { state = 653 } #Yekaterinburg
					ROOT = { controls_province = 1821 }
				}
				ROOT = {
					country_event = tram.309
				}
			}
			
			# RUS-FIN peace
			if = {
				limit = {
					OR = {
						is_russian_tag = yes
						tag = FIN
					}
					has_war_with = FROM
					FROM = {
						OR = {
							is_russian_tag = yes
							tag = FIN
						}
					}
				}
				FIN = { country_event = rusfor.1232 }
			}
			if = {
				limit = {
					OR = {
						is_russian_tag = yes
						tag = ROM
					}
					has_war_with = FROM
					FROM = {
						OR = {
							is_russian_tag = yes
							tag = ROM
						}
					}
				}
				ROM = { country_event = rusfor.1236 }
			}
		}
	}
	on_daily_SOV = {
		effect = {
			if = {
				limit = {
					has_dynamic_modifier = { modifier = SOV_cultural_influence_modifier }
				}
				SOV_cultural_recheck_colonies = yes
				SOV_daily_update_cultural = yes
				SOV_update_cultural_effects = yes
			}
		}
	}
	on_government_change = {
		effect = {
			if = {
				limit = {
					tag = TRM
					OR = {
						has_government = social_liberal
						has_government = social_democrat
					}
				}
				set_cosmetic_tag = TRM_SIB
				if = {
					limit = {
						country_exists = SIB
						has_global_flag = TRM_civil_war
					}
					hidden_effect = {
						TRM = {
							every_unit_leader = {
								retire = yes
							}
						}
					}
					SIB = {
						every_unit_leader = {
							set_nationality = TRM
						}
					}
					TRM = {
						annex_country = { target = SIB transfer_troops = yes }
						add_stability = 0.1
						add_war_support = -0.1
						clr_country_flag = TRM_Yui
						clr_country_flag = TRM_Moryaki_Coup
						clr_country_flag = TRM_NRA_coup
						clr_country_flag = TRM_Semyonov_Coup
						clr_country_flag = TRM_Kolchak_alive
					}
				}
			}
			if = {
				limit = {
					tag = TRM
					NOT = {
						OR = {
							has_government = social_liberal
							has_government = social_democrat
						}
					}
					has_cosmetic_tag = TRM_SIB
				}
				drop_cosmetic_tag = yes
			}
			if = {
				limit = {
					tag = TRM
					NOT = {
						has_country_leader = {
							name = "Aleksandr Kolchak"
							ruling_only = yes
						}
					}
				}
				if = {
					limit = {
						country_exists = BRY
						BRY = { is_subject_of = TRM }
					}
					BRY = {
						country_event = {
							id = bry.50
							days = 4
						}
					}
				}
			}
		}
	}
				
}
