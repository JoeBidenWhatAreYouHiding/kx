COL_border_states_dynamic_mod = {
	icon = "GFX_population_icon"
	enable = { always = yes }
	remove_trigger = { 
		OR = {
			OR = {
				NOT = { is_owned_by = COL }
				any_neighbor_state = {
					OR = {
						NOT = { is_fully_controlled_by = PRU }
						PRU = { 
							OR = { 
								NOT = { can_declare_war_on = COL }
								AND = {
									has_socialist_government = yes
									COL = { has_socialist_government = yes }
								}
							}
						}
					}
				}
				any_neighbor_state = {
					OR = {
						NOT = { is_fully_controlled_by = ECU }
						ECU = { 
							OR = { 
								NOT = { can_declare_war_on = COL }
								AND = {
									has_socialist_government = yes
									COL = { has_socialist_government = yes }
								}
							}
						}
					}
				}
			}
			COL = {
				has_completed_focus = COL_reopening_the_panamerican_road
			}
		}
	}	
	state_production_speed_industrial_complex_factor = -0.1
	state_production_speed_arms_factory_factor = -0.1
	state_production_speed_infrastructure_factor = -0.1
	local_factories = -0.15
	local_resources = -0.25
	mobilization_speed = -0.25
}
