	#Anarchists activities ------------------------------- 
	#change this so momentum builds, with a chance of an activity that resets momentum, higher momentum = more destructive
	#higher momentum = slower build but moves bop more
	#lower momentum has higher chances of just building more when the timer runs out instead of doing a weak activity
	ENG_Anarchist_activity_mission = {
		activation = {
			has_power_balance = {
				id = ENG_Anarchist_Totalist_balance
			}
		}
		available = {
			always = no
		}
		cancel_if_not_visible = yes
		visible = {
			has_government = totalist
			has_power_balance = {
				id = ENG_Anarchist_Totalist_balance
			}
		}
		icon = generic_ignite_civil_war
		is_good = no
		days_mission_timeout = 45
		days_re_enable = 7
		selectable_mission = no
		timeout_effect = {
			if = {
				limit = {
					check_variable = {
						var = ENG.ENG_anarchist_momentum
						value = 4
						compare = less_than_or_equals
					}
				}
				ENG_anarchist_do_activity_low = yes
			}
			else_if = {
				limit = {
					AND = {
						check_variable = {
							var = ENG.ENG_anarchist_momentum
							value = 5
							compare = greater_than_or_equals
						}
						check_variable = {
							var = ENG.ENG_anarchist_momentum
							value = 7
							compare = less_than_or_equals
						}
					}
				}
				ENG_anarchist_do_activity_med = yes
			}
			else_if = {
				limit = {
					check_variable = {
						ENG.ENG_anarchist_momentum > 7
					}
				}
				ENG_anarchist_do_activity_high = yes
			}
		}
	}
