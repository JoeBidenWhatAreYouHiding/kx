### Faction Initiative Spending Priorities ###

### Facilities ###
faction_add_facility = {
	allowed = { has_dlc_ncns = yes }
	enable = {
		is_actual_major = yes # Only majors should ever assign their facilities
	}
	abort_when_not_enabled = yes
	ai_strategy = {
		type = spent_faction_initiative_priority
		id = program
		value = 10 # Moderate Priority as not that important
	}
}

### Military Operations ###
faction_create_faction_theatres = {
	allowed = { has_dlc_ncns = yes }
	enable = {
		always = yes
	}
	abort_when_not_enabled = yes
	ai_strategy = {
		type = spent_faction_initiative_priority
		id = unlock_faction_commander
		value = 30 #AI will heavily prioritise creating faction theatres. Makes doctrine sharing more powerful.
	}
}
