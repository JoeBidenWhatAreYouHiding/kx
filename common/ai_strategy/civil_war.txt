canada_war_with_csa = {
	allowed = {
		NOT = {
			tag = CAN
			tag = MEX
			tag = BEL
		}
	}
	enable = {
		is_in_faction_with = CAN
		num_divisions > 4
		OR = {
			has_war_with = CSA
			has_war_with = TEX
			has_war_with = USA
			has_war_with = CAL
		}
		NOT = {
			surrender_progress > 0
			AND = {
				has_war_with = SRI
				is_italian_tag = yes
			}
			AND = {
				has_war_with = FRA
				tag = CAR
			}
		}
	}

	abort_when_not_enabled = yes

	ai_strategy = {
		type = protect
		id = "CAN"
		value = 200
	}
	ai_strategy = {
		type = send_volunteers_desire
		id = "CAN"
		value = 10000
	}
	ai_strategy = {
		type = force_defend_ally_borders
		id = "CAN"
		value = 1000
	}
	ai_strategy = {
		type = area_priority
		id = canada_america
		value = 500
	}
}
