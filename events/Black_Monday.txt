### Black Monday Event for Germany
country_event = {
	id = black.2
	immediate = { 
		log = "[GetDateText]: [Root.GetName]: event black.2" 
		hidden_effect = {
			set_global_flag = black_monday
			add_stability = -0.50
			add_ideas = GER_black_monday
			subtract_from_variable = {
				var = GER.GER_mitteleuropa_strength_var
				value = 40
				tooltip = sub_40_from_mitteleuropastrength_tt
			}
			GER_recalculate_mitteleuropa_effects = yes
			country_event = {
				id = black.7
				days = 35
			}
			country_event = {
				id = gerdom.17
				days = 4
			}
			country_event = {
				id = gerdom.11
				days = 40
			}
			country_event = {
				id = ethiopia.2
				days = 45
			}
		}
	}
	title = black.2.t
	desc = black.2.d
	#picture = GFX_news_event_GER_blackmonday1 # Needs country event one
	is_triggered_only = yes

	option = {
		name = black.2.a
		effect_tooltip = {
			add_stability = -0.50
			add_ideas = GER_black_monday

				var = GER.GER_mitteleuropa_strength_var

			}
		}
	}
}

country_event = {
	id = black.7
	immediate = { log = "[GetDateText]: [Root.GetName]: event black.7" }

	desc = black.7.d

	is_triggered_only = yes

	option = {
		name = black.7.a
		custom_effect_tooltip = GER_colonial_resources_distribution_full_tt
		add_political_power = -20
		set_country_flag = GER_helped_colonies_full
		add_to_variable = {
			var = GER.GER_mitteleuropa_strength_var
			value = 2
			tooltip = add_2_to_mitteleuropastrength_tt
		}
		add_timed_idea = {
			idea = GER_colonial_subsidies

		}

			limit = {

				}
