	immediate = {log = "[GetDateText]: [Root.GetName]: event mexico.15"}
	title = mexico.15.t
	desc = mexico.15.d
	picture = diplomacy2

	trigger = {
		tag = MEX
		has_socialist_government = yes
		has_idea = MEX_cristero_war_idea
		NOT = {
			has_idea = MEX_cristero_attack
		}
		any_owned_state = {
			is_core_of = MEX
			industrial_complex > 0
			is_controlled_by = MEX
		}
	}

	mean_time_to_happen = {
		months = 2
		modifier = {
			factor = 0.5
			has_completed_focus = MEX_socialist_education
		}
		modifier = {
			factor = 1.5
			has_completed_focus = MEX_state_atheism
		}
	}
	immediate = {
		random_owned_state = {
			limit = {
				is_core_of = MEX
				industrial_complex > 0
				is_controlled_by = MEX
			}
			save_event_target_as = attack_state
		}
	}

	option = {
		name = mexico.15.a
		add_timed_idea = {
			idea = MEX_cristero_attack
			days = 90
		}
		add_stability = -0.01
		event_target:attack_state = {
			damage_building = {
				type = industrial_complex
				damage = 1.0
			}
		}
	}
}
