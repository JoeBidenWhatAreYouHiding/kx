	immediate = { log = "[GetDateText]: [Root.GetName]: event tur.92 FROM = [From.GetName]" }
	title = tur.92.t
	desc = tur.92.d
	picture = GFX_OTT_Barzani_Revolt

	fire_only_once = yes

	trigger = {
		tag = TUR
		OR = {
			AND = {
				check_variable = {
					kurdish_situation > 99
				}
				check_variable = {
					kurdish_round > 7
				}
			}
			check_variable = {
				kurdish_situation > 199
			}
		}
	}

	option = {
		name = tur.92.a
		clear_variable = kurdish_round
		activate_mission = TUR_kurdish_revolt_timeout
		354 = {
			set_demilitarized_zone = no
		}
		1000 = {
			set_demilitarized_zone = no
		}
		add_timed_idea = {
			idea = TUR_kurdish_crisis
			days = 100
		}
		hidden_effect = {
			remove_decision = TUR_literacy_campaign_in_kurdistan
			remove_decision = TUR_hunt_down_rebels
			remove_decision = TUR_send_the_army
			remove_decision = TUR_arrest_leaders
			remove_decision = TUR_increase_police_presence
			remove_decision = TUR_grant_autonomy_to_kurdistan
		}
		### Anatolian Kurdistan
		every_owned_state = {
			limit = {
				OR = {
					state = 350
					state = 352
					state = 354
					state = 1000
				}
			}
			KDR = {
				add_state_core = PREV
				transfer_state = PREV
			}
		}
		if = {
			limit = { country_exists = KDR }
			KDR = {
				load_oob = KDR_revolt
				add_ideas = KDR_islamic_fervour
				declare_war_on = {
					target = TUR
					type = annex_everything
				}
				set_politics = {
					ruling_party = paternal_autocrat
					elections_allowed = no
				}
				#load_focus_tree = kdr_focus
				#complete_national_focus = KDR_kurdish_revolt
				clear_variable = kurdish_situation
				remove_mission = TUR_kurdish_sentiment_increase
			}
		}
	}
}
