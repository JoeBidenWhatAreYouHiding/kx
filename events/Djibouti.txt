		}

	}

			id = djibouti.2

		}

		}

			add_idea = DJI_new_nationhood
		}
		AFR_open_political_tree = yes
		AFR_open_economy_tree = yes
	}
	
	option = { #Russians
		name = djibouti.1.c
		country_event = {
			id = djibouti.3
			days = 45
		}
		DKB = {
			every_unit_leader = {
				limit = {
					has_id = 42021
				}
				set_nationality = DJI
			}
		}
		create_country_leader = {
			name = "Semyon Krasnov"
			desc = "POLITICS_SEMEN_KRASNOV_DESC"
			picture = "gfx/leaders/DKB/Portrait_DKB_Semyon_Krasnov.png"
			expire = "1965.1.1"
			ideology = autocratic_colonial_govt_subtype
			traits = {}
		}
		if = {
		limit = {
		is_ai = no
		}
		create_wargoal = {
			type = annex_everything
			target = ETH
		}
		create_wargoal = {
			type = annex_everything
			target = SOM
		}
		create_wargoal = {
			type = annex_everything
			target = ERI
		}
		every_state = {
			limit = {
				OR = {
					is_core_of = ETH
					is_claimed_by = ETH
					is_core_of = SOM
					is_claimed_by = SOM
					is_core_of = ERI
					is_claimed_by = ERI
				}
			}
			add_core_of = DJI
			}
		}
		set_country_flag = DJI_SAGALLO_FORMED
		set_cosmetic_tag = DJI_RUS
		swap_ideas = {
			remove_idea = DJI_chaos_n_shit
			add_idea = DJI_new_sagallo
		}
		set_party_name = {
			name = DJI_paternal_autocrat_party2
			long_name = DJI_paternal_autocrat_party2_long
			ideology = paternal_autocrat
		}
		AFR_open_political_tree = yes
		AFR_open_economy_tree = yes
	}
	
	option = { #Frogs
		name = djibouti.1.e
		DJI = { get_current_government_type = yes }
		NFA = {
			puppet = DJI

			ruling_party = authoritarian_democrat
			elections_allowed = yes
		}
		AFR_open_political_tree = yes
		AFR_open_economy_tree = yes
	}
