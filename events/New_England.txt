country_event = { # Occupy New England
	id = newengland.1
	immediate = {log = "[GetDateText]: [Root.GetName]: event newengland.1"}
	title = newengland.1.t
	desc = newengland.1.d
	picture = GFX_USA_Riots

	is_triggered_only = yes

	option = {
		name = newengland.1.a
		ai_chance = { # Take New England
			factor = 100
		}
		every_state = {
			limit = {
				is_owned_by = USA
				OR = {
					state = 357 #Maine
					state = 934 #New Hampshire
					state = 933 #Vermont
					state = 932 #Massachusetts
					state = 935 #Rhode Island
					state = 931 #Connecticut
					state = 796 #Upstate New York
				}
			}
			CAN = {
				transfer_state = PREV
			}
		}
		every_country = {
			limit = {
				exists = yes
				OR = {
					tag = USA
					tag = CSA
					tag = TEX
					tag = CAL
				}
			}
			country_event = {
				id = newengland.2
				days = 1
			}
		}
	}
	option = {
		name = newengland.1.b
		ai_chance = {
			factor = 0
		}
		add_stability = -0.01
	}
}
