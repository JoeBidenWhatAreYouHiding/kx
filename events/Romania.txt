			set_state_controller = 1117 #North Transylvania
			set_state_controller = 1118 #North Transylvania
			set_state_controller = 789 #Szeklerland

				limit = {
					check_variable = { ROM.revolt > 8 }
				}

					add_timed_idea = {
						idea = ROM_TRS_revolt
						days = 30
					}
				}
			}

					check_variable = { ROM.revolt > 6 }
				}
				load_oob = "ROM_TRS_revolt1"

					add_timed_idea = {
						idea = ROM_TRS_revolt
