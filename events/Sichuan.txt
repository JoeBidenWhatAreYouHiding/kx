	title = sichuan.12.t

	desc = {
		text = sichuan.12.d.SZC

	}
	desc = {

		trigger = { tag = TIB }
	}

	picture = GFX_report_event_TIB_army

	fire_only_once = yes
	is_triggered_only = yes

	trigger = {
		TIB = {
			owns_state = 693
			has_war_with = SZC
			is_valid_peace_target = yes
		}
		SZC = {
			owns_state = 601
			is_valid_peace_target = yes
		}
		if = {
			limit = { tag = TIB }
			surrender_progress > 0.25
			SZC = {
				surrender_progress < 0.1
				controls_state = 693 #Chamdo
				controls_state = 748 #Bome
			}
		}
		else = {
			TIB = {
				controls_state = 601 #Xikang
				surrender_progress < 0.1
			}
			OR = {
				surrender_progress > 0.2
				TIB = { controls_state = 788 } #Aba
			}
