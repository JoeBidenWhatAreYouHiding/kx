		has_country_leader = {
			name = "Douglas MacArthur"
			ruling_only = yes

		}
		has_government = paternal_autocrat
		has_global_flag = NO_ANIMATED_PORTRAITS
	}
	hidden = yes
	fire_only_once = yes

}
