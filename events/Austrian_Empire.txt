	title = constituents.1.t
	desc = constituents.1.d

	is_triggered_only = yes
	
	trigger = {
		OR = {
			check_variable = { AUS.subject_revolt_progress = 1 }
			check_variable = { AUS.subject_revolt_progress > 1 }
		}
		tag = AUS
		NOT = {
			OR = {
				has_global_flag = no_more_collapse
				has_global_flag = AUS_collapsed
				has_country_flag = imminent_collapse
				has_active_mission = BOTDE_collapse_headsup 
			}
		}
	}
	
	immediate = {
		hidden_effect = {
			every_owned_state = {
				clr_state_flag = rev_point
			}
			if = {
				limit = {
					NOT = {
						has_global_flag = AUS_rev_point_schutzbund
					}
				}
				random_owned_state = {
					limit = {
						NOT = {
							has_state_flag = rev_point
						}
						has_state_flag = AUS_schutzbund
					}
					add_to_variable = { AUS.schutzbund_rev_point_var = 0.5 }
					set_state_flag = rev_point
				}
				random_owned_state = {
					limit = {
						NOT = {
							has_state_flag = rev_point
						}
						has_state_flag = AUS_schutzbund
					}
					add_to_variable = { AUS.schutzbund_rev_point_var = 0.5 }
					set_state_flag = rev_point
				}
			}
			if = {
				limit =  {
					check_variable = { schutzbund_rev_point_var = 1 }
				}
				add_to_variable = { revolution_value = 1 }
				set_global_flag = AUS_rev_point_schutzbund
			}
		}
	}
	
	option = { #Gott erhalte, gott beschutze...
		name = constituents.1.a
		activate_mission = BOTDE_collapse_headsup
		if = {
			limit = {
				HUN = { is_in_faction_with = AUS }
				check_variable = { HUN.subject_revolt_progress < 0.75 }
			}
			set_country_flag = AUS_collapse_HUN_stays
		}		
		#### Collapse type		
		# Revolution		
		if = {
			limit = {
				check_variable = { revolution_value > 3 }
			}
			set_country_flag = AUS_revolution_collapse
			country_event = {
