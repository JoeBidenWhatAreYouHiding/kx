### Ask for the Iron Crown of Lombardy
country_event = {
	id = lombardy.1
	immediate = { log = "[GetDateText]: [Root.GetName]: event lombardy.1" }
	title = lombardy.1.t
	picture = GFX_report_event_ITA_milan

	desc = {
		text = lombardy.1.d.aus
		trigger = { country_exists = AUS }
	}
	desc = {
		text = lombardy.1.d.sic
		trigger = {
			NOT = { country_exists = AUS }
			country_exists = SIC
		}
	}
	desc = {
		text = lombardy.1.d.vienna
		trigger = {
			NOT = { country_exists = AUS }
			NOT = { country_exists = SIC }
		}
	}

	is_triggered_only = yes

	option = {
		name = lombardy.1.a
		trigger = { country_exists = AUS }
		AUS = {
			country_event = lombardy.2
		}
	}
	option = {
		name = lombardy.1.b
		trigger = {
			NOT = { country_exists = AUS }
			country_exists = SIC
		}
		SIC = {
			country_event = lombardy.2
		}
	}
	option = {
		name = lombardy.1.c
		trigger = {
			NOT = { country_exists = AUS }
			NOT = { country_exists = SIC }
		}
		random_country = {
			limit = { owns_state = 4 }
			country_event = lombardy.2
		}
	}
}

### Lombardy asks for the Iron Crown
country_event = {
	id = lombardy.2
	immediate = { log = "[GetDateText]: [Root.GetName]: event lombardy.2" }
	title = lombardy.2.t
	desc = lombardy.2.d
	picture = GFX_report_event_ITA_milan

	is_triggered_only = yes

	#Return it
	option = {
		name = lombardy.2.a
		ai_chance = {
			factor = 90
			modifier = {
				factor = 0
				has_socialist_government = yes
			}
		}
		add_political_power = -10
		FROM = {
			country_event = {
				id = lombardy.3
				days = 3
			}
		}
		LOM = {
			add_opinion_modifier = {
				target = ROOT
				modifier = LOM_returned_crown
			}
		}
	}

	#Refuse!
	option = {
		name = lombardy.2.b
		trigger = {
			LOM = {
				is_ally_of_ROOT = no
			}
		}
		ai_chance = {
			factor = 10
		}
		FROM = {
			country_event = {
				id = lombardy.4
				days = 3
			}
		}
		LOM = {
			add_opinion_modifier = {
				target = ROOT
				modifier = LOM_refused_crown
			}
		}
	}
}

### Hire Military Advisors
country_event = {
	id = lombardy.6
	immediate = { log = "[GetDateText]: [Root.GetName]: event lombardy.6" }
	title = lombardy.6.t
	desc = lombardy.6.d

	is_triggered_only = yes

	option = { # Prussian advisors are the best
		trigger = {
			country_exists = GER
			NOT = {
				has_war_with = GER
			}
		}
		ai_chance = {
			factor = 50
			modifier = {
				if = {
					limit = {
						country_exists = GER
						OR = {
							is_in_faction_with = GER
							is_subject_of = GER
						}
					}
				}
				factor = 10
			}
		}
		name = lombardy.6.a
		add_ideas = LOM_Prussian_drill
		if = {
			limit = {
				country_exists = GER
			}
			add_opinion_modifier = {
				target = GER
				modifier = LOM_prussian_instructors
			}
			hidden_effect = {
				GER = {
					country_event = lombardy.7
				}
			}
		}
	}
	option = { # Italian advisors are popular
		ai_chance = {
			factor = 50
		}
		name = lombardy.6.b
		add_political_power = 10
		add_ideas = LOM_Italian_drill
	}
}
