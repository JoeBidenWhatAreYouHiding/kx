	id = delhi.5
	immediate = { log = "[GetDateText]: [Root.GetName]: event delhi.5" }
	title = delhi.5.t
	desc = delhi.5.d
	picture = GFX_report_event_PRF_war

	fire_only_once = yes

	trigger = {
		tag = DEH
		has_war_with = AFG
		has_country_flag = DEH_afghan_war
		controls_state = 442 #Peshawar
		controls_state = 445 #Quetta
		NOT = {
			any_owned_state = {
				is_controlled_by = AFG
			}
		}
		AFG = {
			owns_state = 442 #Peshawar
			owns_state = 445 #Quetta
		}
	}

	mean_time_to_happen = {
		days = 5
	}

	option = { # take cores
		name = delhi.5.a
		ai_chance = {
			factor = 50
		}
		AFG = {
			remove_state_core = 442 #Peshawar
			remove_state_core = 445 #Quetta
			remove_state_core = 1209
			remove_state_claim = 444 #Baluchistan
			white_peace_with_ROOT = yes
		}
		DEH = {
			transfer_state = 442 #Peshawar
			transfer_state = 445 #Quetta
			transfer_state = 1209
		}
		hidden_effect = {
			AFG = {
				country_event = afg.1
				set_country_flag = AFG_lose_war
				country_event = {
					id = afg.8
					days = 7
					random = 864
				}
			}
		}
	}
	option = { # puppet
		name = delhi.5.b
		trigger = {
			controls_state = 267 #Kabul
		}
		ai_chance = {
			factor = 50
		}
		add_political_power = -100
		AFG = {
			remove_state_core = 442 #Peshawar
			remove_state_core = 445 #Quetta
			remove_state_core = 1209
			white_peace_with_ROOT = yes
		}
		DEH = {
			transfer_state = 442 #Peshawar
			transfer_state = 445 #Quetta
			transfer_state = 1209
			puppet = AFG
		}
		hidden_effect = {
			country_event = {
				id = delhi.6
				days = 1
			}
			AFG = {
				country_event = afg.1
			}
		}
	}
	option = { #conquest
		name = delhi.5.c
		ai_chance = {
			factor = 0
		}
		add_political_power = -150
		add_stability = -0.05
	}
}
