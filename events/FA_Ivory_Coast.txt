        id = civ.1
        immediate = {log = "[GetDateText]: [Root.GetName]: event civ.1"}
        title = civ.1.t
	desc = civ.1.d
        picture = GFX_report_event_CIV_generic

	fire_only_once = yes

	trigger = {
        tag = CIV
		NOT = {
			has_country_flag = CIV_freefrombitches
		}
		country_exists = TZN	
		OR = {
			has_country_flag = CIV_bout_to_explode
			has_government = paternal_autocrat
		}
    }

	immediate = {
		hidden_effect = {
			every_unit_leader = {
				limit = {
					has_id = 12434 
				}
				demote_leader = yes
			}
			every_unit_leader = {
				limit = {
					has_id = 12733 
				}
				promote_leader = yes
			}

			GHA = { 
				transfer_state = 274		
			}
		
			if = {
				limit = {
					country_exists = GHA
					GHA = { NOT = { has_country_flag = MAF_already_exists } }
				}
				GHA = {
					add_state_core = 274
					load_oob = GHA
				}
			}
			if = {
				limit = {
					owns_state = 807
				}
				NFA = { 
					transfer_state = 807
					transfer_state = 963 		
				}
			}
			
			if = {
				limit = {
					owns_state = 876
				}
				DAH = { transfer_state = 876 }
			}
		
			NFA = {
				puppet = DAH
			}
			if = {
				limit = {
					country_exists = DAH
					DAH = { NOT = { has_country_flag = MAF_already_exists } }
				}
				DAH = {
					add_state_core = 876
					load_oob = DAH
				}
			}
		}
	}

	option = {
		name = civ.1.a
		ai_chance = {
			factor = 100
		}
		TOG = {
			annex_country = {
				target = CIV
				transfer_troops = yes
			}
			add_state_core = 877
		}
		TOG = {
			change_tag_from = CIV
		}
		every_unit_leader = { set_nationality = TOG }
	}
}
