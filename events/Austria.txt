		name = aus.elections.100.a

		hidden_effect = {

				limit = {

						has_game_rule = {
							rule = AUS_start_election
							option = AUS_START_PATERNAL_AUTOCRAT
						}
					}
				}
				country_event = {
					id = aus.elections.101
					days = 1
				}
			}
			else_if = {
				limit = {
					OR = {
						has_game_rule = {
							rule = AUS_start_election
							option = AUS_START_SOCIAL_DEMOCRAT
						}
						has_game_rule = {
							rule = AUS_start_election

						}

						}
					}
				}
				country_event = {
					id = aus.elections.102
					days = 1
				}
			}
			else = {
				if = {
					limit = {
						social_conservative_plurality = yes
					}
					country_event = {
						id = aus.elections.101
