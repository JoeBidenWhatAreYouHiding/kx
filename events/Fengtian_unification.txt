country_event = { #hidden conference starter event
	id = fngshen.0
	immediate = { log = "[GetDateText]: [Root.GetName]: event fngshen.0" }
	hidden = yes

	trigger = {
		NOT = { has_country_flag = FNG_shenyang_conference }
		NOT = { has_country_flag = FNG_conference_fired }
		OR = {
			ANQ = {
				exists = yes
				is_ruled_by_pro_fengtian = yes
			}
			GXC = {
				exists = yes
				GXC_has_federalist_government = yes
			}
			PRC = {
				exists = yes
				GXC_has_federalist_government = yes
			}
			SZC = {
				exists = yes
				SZC_federalists_in_exile = yes
			}
		}
		QIE = {
			exists = yes
		}
	}

	is_triggered_only = yes

	option = { #war already started, PRC-GXC not finished -> only for Anqing
		trigger = {
			has_war_with = QIE
			ANQ = {
				exists = yes
				is_ruled_by_pro_fengtian = yes
				has_country_flag = ANQ_peace
				NOT = {
					has_country_flag = ANQ_feng_aid_undecided
				}
			}
		}
		set_country_flag = FNG_shenyang_anq_only
		country_event = {
			id = fngshen.1
			days = 1
			random_hours = 12
		}
	}

	option = { #wait for the GXC-PRC war to end and ANQ to have peace / SQI won League War and FNG did national irredentism
		trigger = {
			OR = {
				GXC = {
					GXC_has_federalist_government = yes
					has_war_with = PRC
				}
				PRC = {
					GXC_has_federalist_government = yes
					has_war_with = GXC
				}
				ANQ = {
					exists = yes
					is_ruled_by_pro_fengtian = yes
					NOT = { has_country_flag = ANQ_peace }
				}
				NOT = { has_country_flag = FNG_national_irredentism_flag }
			}
		}
		country_event = {
			id = fngshen.0
			days = 5
			random = 12
		}
	}

	option = { #ANQ won League War and the GXC-PRC war is finished
		trigger = {
			has_country_flag = FNG_national_irredentism_flag
			ANQ = {
				exists = yes
				is_ruled_by_pro_fengtian = yes
				has_country_flag = ANQ_peace
			}
			NOT = {
				GXC = {
					GXC_has_federalist_government = yes
					has_war_with = PRC
				}
				PRC = {
					GXC_has_federalist_government = yes
					has_war_with = GXC
				}
			}
		}
		country_event = fngshen.1
	}

	option = { #ANQ is not pro fengtian / does not exist
		trigger = {
			has_country_flag = FNG_national_irredentism_flag
			NOT = {
				ANQ = {
					exists = yes
					is_ruled_by_pro_fengtian = yes
				}
				GXC = {
					GXC_has_federalist_government = yes
					has_war_with = PRC
				}
				PRC = {
					GXC_has_federalist_government = yes
					has_war_with = GXC
				}
			}
		}
		set_country_flag = FNG_shenyang_no_allies
		country_event = fngshen.4
	}
}

country_event = { #A Conference in Shenyang
	id = fngshen.1
	immediate = { log = "[GetDateText]: [Root.GetName]: event fngshen.1" }
	title = fngshen.1.t
	desc = {
		text = fngshen.1.d1
		trigger = {
			NOT = { has_country_flag = FNG_shenyang_anq_only }
		}
	}
	desc = {
		text = fngshen.1.d2
		trigger = {
			has_country_flag = FNG_shenyang_anq_only
		}
	}

	picture = GFX_report_event_CHI_new_government

	is_triggered_only = yes

	trigger = {
		NOT = { has_country_flag = FNG_conference_fired }
	}

	immediate = {
		hidden_effect = {
			set_country_flag = FNG_shenyang_conference
		}
	}

	option = {
		name = fngshen.1.a #Send the invites
		hidden_effect = {
			ANQ = {
				if = {
					limit = { is_ruled_by_pro_fengtian = yes }
					country_event = {
						id = fngshen.2
						days = 1
						random = 14
					}
					FNG = { set_country_flag = FNG_shenyang_ANQ } #they autoaccept
				}
			}
			if = {
				limit = {
					has_country_flag = FNG_shenyang_anq_only
				}
				country_event = {
					id = fngshen.7
					days = 5
				}
			}
			else = {
				SQI = {
					if = {
						limit = { is_ruled_by_pro_fengtian = yes }
						country_event = {
							id = fngshen.2
							days = 1
							random = 14
						}
						FNG = { set_country_flag = FNG_shenyang_SQI } #they autoaccept
					}
				}
				if = {
					limit = {
						OR = {
							GXC = {
								exists = yes
								GXC_has_federalist_government = yes
								NOT = { has_war_with = ROOT }
							}
							SZC = {
								exists = yes
								SZC_federalists_in_exile = yes
								NOT = { has_war_with = ROOT }
							}
						}
					}
					country_event = {
						id = fngshen.4
						days = 8
					}
				}
				else = { #if there are no federalists anymore
					country_event = {
						id = fngshen.7
						days = 8
					}
				}
			}
		}
	}
}
