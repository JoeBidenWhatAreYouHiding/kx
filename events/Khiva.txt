unit_leader_event = { #Death of Junaid Khan
	id = khiva.2	
	title = khiva.2.t
	desc = khiva.2.d

	fire_only_once = yes
	is_triggered_only = yes
	trigger = {
		KHI = { KHI_junaid_khan_in_power = yes }
	}

	option = {
		name = khiva.2.a
		trigger = { check_variable = { KHI.eshi_khan_influence > 4 } }
		KHI = {
			remove_ideas_with_trait = head_of_government
			random_unit_leader = {
				limit = { has_id = 68589 }			
				remove_unit_leader = yes
			}			
			create_country_leader = {
				name = "Eshi-Khan"
				picture = "gfx/leaders/KHI/Portrait_KHI_Eshi_Khan.png"
				expire = "1965.1.1"
				ideology = pataut_warlordism_subtype
			}
			if = {
				limit = {
					OR = {
						has_cosmetic_tag = KHI_new_khanate
						has_cosmetic_tag = KHI_seljuk
						has_cosmetic_tag = KHI_seljuk2
						has_cosmetic_tag = KHI_seljuk3
						has_cosmetic_tag = KHI_khwarazm
					}
				}
				set_country_leader_ideology = absolute_monarchy_subtype
			}
			else = {
				set_party_name = {
					ideology = paternal_autocrat
					name = KHI_paternal_autocrat_party2
					long_name = KHI_paternal_autocrat_party2_long
				}
			}
			if = {
				limit = { has_completed_focus = KHI_Establish_Turkmenbashi }
				add_country_leader_trait = KHI_turkmenbashi
			}
		}
	}
	option = {
		name = khiva.2.b
		trigger = { check_variable = { KHI.eshi_khan_influence < 5 } }
		KHI = {
			drop_cosmetic_tag = yes
			remove_ideas_with_trait = head_of_government
			random_unit_leader = {
				limit = { has_id = 68589 }			
				remove_unit_leader = yes
			}
			set_politics = {
				ruling_party = national_populist
				elections_allowed = no
			}
			if = {
				limit = { has_completed_focus = KHI_Establish_Turkmenbashi }
				add_country_leader_trait = KHI_turkmenbashi
			}
			if = {
				limit = {
					has_completed_focus = KHI_Legacy_Of_Junaid_Khan
				}
				uncomplete_national_focus = {
					focus = KHI_Legacy_Of_Junaid_Khan
					uncomplete_children = yes
				}
				complete_national_focus = KHI_Bury_Khivan_Monarchy
			}
			every_state = {
				limit = {
					is_claimed_by = KHI
					NOT = { is_owned_by = KHI }
				}
				remove_claim_by = KHI
			}
		}
	}
}
