country_event = { #Germany Proposes a Single Currency
	id = germit.1
	immediate = { log = "[GetDateText]: [Root.GetName]: event germit.1" }
	title = germit.1.t

	desc = {
		text = germit.1.d.independent
		trigger = { is_subject = no }
	}
	desc = {
		text = germit.1.d.subject
		trigger = { is_subject = yes }
	}

	picture = GFX_report_event_generic_read_write

	trigger = {
		GER_is_in_mitteleuropa = yes
	}

	is_triggered_only = yes

	option = { # Yes
		name = germit.1.a
		ai_chance = {
			factor = 80
			modifier = {
				factor = 0
				has_game_rule = {
					rule = GER_mitteleuropa
					option = GER_MITTELEUROPA_NOT_STONKS
				}
			}
			modifier = {
				factor = 0
				tag = ROM
				is_subject = no
			}
		}
		add_political_power = -25
		swap_ideas = {
			remove_idea = GER_mitteleuropa_0
			add_idea = GER_mitteleuropa_1
		}
		if = {
			limit = { is_subject = no }
			GER = {
				country_event = { id = germit.2 days = 1 random_days = 3 }
				}
			}
		if = {
			limit = {
				has_idea = DEN_idea_weak_krona
		}
			remove_ideas = DEN_idea_weak_krona
			}
		hidden_effect = {
			every_subject_country = { country_event = germit.1 }
		}
	}

	option = { # No
		name = germit.1.b
		trigger = { is_subject = no }
		ai_chance = {
			factor = 20
			modifier = {
				factor = 0
				has_game_rule = {
					rule = GER_mitteleuropa
					option = GER_MITTELEUROPA_STONKS
				}
			}
		}
		add_political_power = 25
		GER = {
			country_event = { id = germit.3 days = 1 random_days = 3 }
		}
	}
}

country_event = { #Leaving Mitteleuropa
	id = germit.4
	immediate = { log = "[GetDateText]: [Root.GetName]: event germit.4" }
	title = germit.4.t

	desc = {
		text = germit.4.d.GER_is_dead
		trigger = {
			NOT = { country_exists = GER }
		}
	}
	desc = {
		text = germit.4.d.socialist
		trigger = {
			country_exists = GER
			has_socialist_government = yes
		}
	}
	desc = {
		text = germit.4.d.war_with_GER
		trigger = {
			has_socialist_government = no
			has_war_with = GER
		}
	}
	desc = {
		text = germit.4.d.allied_with_rival
		trigger = {
			has_socialist_government = no
			NOT = { has_war_with = GER }
			AND = {
				is_in_faction = yes
				NOT = { is_in_faction_with = GER }
			}
		}
	}
	desc = {
		text = germit.4.d.left_mitteleuropa
		trigger = {
			has_socialist_government = no
			NOT = {
				has_war_with = GER
				has_country_flag = GER_kicked_from_mitteleuropa
			}
			is_in_faction = no
		}
	}
	desc = {
		text = germit.4.d.kicked_out
		trigger = {
			has_country_flag = GER_kicked_from_mitteleuropa
		}
	}

	picture = GFX_report_event_generic_read_write

	is_triggered_only = yes

	trigger = {
		GER_is_in_mitteleuropa = yes
	}

	option = {
		name = germit.4.a
		GER_remove_from_mitteleuropa = yes
		if = {
			limit = {
				NOT = {
					has_war_with = GER
					has_country_flag = GER_kicked_from_mitteleuropa
				}
			}
			GER = { country_event = germit.5 }
		}
		clr_country_flag = GER_kicked_from_mitteleuropa
	}
}
