#### 1936 Albanian Elections
country_event = {
	id = albania.3
	title = albania.3.t
	desc = albania.3.d
	picture = GFX_report_event_ALB_Election

	fire_only_once = yes

	trigger = {
		tag = ALB
		date > 1936.6.1
		NOT = {
			has_country_flag = ALB_elections_of_1936_done
		}

	immediate = {
		log = "[GetDateText]: [Root.GetName]: event albania.3"

		name = albania.3.a

			factor = 65
			modifier = {
				factor = 0

						rule = ALB_election

						option = ALB_ELECTION_KING_ZOGU
					}
					has_game_rule = {
						rule = ALB_election
						option = ALB_ELECTION_HOXHA_UTOPIA
					}
					has_game_rule = {
						rule = ALB_election
						option = ALB_ELECTION_HOXHA_ARDLYRIA
					}
					has_game_rule = {
						rule = ALB_election
						option = ALB_ELECTION_HOXHA_ODRYSIA
					}
					has_game_rule = {
						rule = ALB_election
						option = ALB_ELECTION_HOXHA_PELASGIA
					}
					has_game_rule = {
						rule = ALB_election

						rule = ALB_election

					}
				}
			}
		}

		set_country_flag = ALB_bashkimi_victory_flag

		add_popularity = {
			ideology = social_democrat
			popularity = 0.1
		}

		set_politics = {
			ruling_party = social_democrat
			election_frequency = 48
			elections_allowed = yes
		}

		if = {
			limit = {
				has_idea = ALB_polarised_nation_idea
			}
			remove_ideas = ALB_polarised_nation_idea
		}

		remove_all_ministers = yes
		add_ideas = {
			ALB_Fan_Noli_hog_sde
			ALB_Sulejman_Delvina_for_sde
			ALB_Ali_Shefqet_Shkupi_eco_sde
			ALB_Bajram_Curri_sec_sde
		}
	}

	option = { #Progressives win
		name = albania.3.b

		ai_chance = {
			factor = 35
			modifier = {
				factor = 0
				OR = {
					has_game_rule = {
						rule = ALB_election
						option = ALB_ELECTION_SOCDEM
					}
					has_game_rule = {
						rule = ALB_election
						option = ALB_ELECTION_KING_WIED
					}
				}
			}
		}

		set_country_flag = ALB_progressive_victory_flag

		add_popularity = {
			ideology = social_conservative
			popularity = 0.1
		}

		set_politics = {
			ruling_party = social_conservative
			election_frequency = 48

		}

		if = {

				has_idea = ALB_polarised_nation_idea
