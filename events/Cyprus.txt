	title = cyp.1.t
	desc = cyp.1.d

	picture = GFX_OTT_Barzani_Revolt

	trigger = {
		has_war_with = EGY
		owns_state = 183
		GRE = {
			exists = yes
			NOT = {
				has_socialist_government = yes
				is_subject = yes
				has_war_with = TUR
				has_completed_focus = GRE_turkish_thaw
			}
		}
	}

	fire_only_once = yes

	is_triggered_only = yes

	immediate = {
		hidden_effect = {
			183 = {
				force_disable_resistance = { clear = yes }
				teleport_armies = {
					limit = { is_ally_with = TUR }
					to_state = 49
				}
			}
			CYP = {
				add_state_core = 183
				add_state_core = 164
				add_state_core = 187
				transfer_state = 183
				load_oob = CYP
				declare_war_on = {
					target = TUR
					type = annex_everything
				}
				add_offsite_building = {
					type = arms_factory
					level = 2
				}
				country_event = {
					id = cyp.7
					days = 3
				}
			}
			GRE = {
				random_army_leader = {
					limit = {
						has_id = 578
					}
					unit_leader_event = {
						id = cyp.5
					}
				}
			}
			news_event = {
				id = worldnews.424
				days = 2
			}
		}
	}

	option = {
		name = cyp.1.a
		trigger = { is_ai = no }
		CYP = {
			change_tag_from = TUR
		}
		every_state = {
			limit = {
				is_core_of = CYP
			}
			ott_increase_revolt_risk = yes
		}
	}
	option = {
		name = cyp.1.b
		custom_effect_tooltip = ott_gre_spread
		every_state = {
			limit = {
				is_core_of = CYP
			}
			ott_increase_revolt_risk = yes
		}
	}
}
