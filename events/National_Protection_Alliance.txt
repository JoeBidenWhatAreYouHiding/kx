					}
					YUN = {
						exists = yes
						is_subject = no
						is_faction_leader = no
						is_in_faction_with = ROOT
					}
					HNN = {
						exists = yes
						is_subject = no
						is_faction_leader = no
						is_in_faction_with = ROOT
					}
					SZC = {
						exists = yes
						is_subject = no
						is_faction_leader = no
						is_in_faction_with = ROOT
					}
				}
			}
			every_other_country = {
				limit = {
					is_in_faction_with = ROOT
					is_subject = no

					id = npa.10
					days = 1
				}
			}
		}
		else = {
			country_event = {
				id = npa.9
				days = 1
			}
		}
