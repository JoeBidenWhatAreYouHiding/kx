			has_war_with = POL
			has_war_with = GAL 
			has_war_with = AUS
			has_war_with = HUN
		}		
	}

	fire_only_once = yes

	option = { #Victory
		name = wsu.29.a
		set_country_flag = WSU_independent
		clear_all_relations = yes
		if = {
			limit = {
				NOT = { owns_state = 91 }
			}
			transfer_state = 91
			add_state_core = 91
		}
		if = {
			limit = {
				OR = {
					controls_state = 1119
					controls_state = 73
				}
			}
			transfer_state = 1119
			transfer_state = 73
			add_state_core = 1119
			add_state_core = 73 
		}
		if = {
			limit = { controls_state = 80 }
			transfer_state = 80
			add_state_core = 80
		}
		every_enemy_country = {
			limit = {
				OR = {
					tag = ROM
					tag = AUS
					tag = POL
					tag = HUN
					tag = GAL
				}
			}
			country_event = wsu.30
		}
	}
