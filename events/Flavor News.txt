news_event = { #1936 Summer Olympics at Vienna!
	id = flavornews.2
	title = flavornews.2.t
	desc = flavornews.2.d
	picture = GFX_NEWS_vienna_olympics

	major = yes

	is_triggered_only = yes

	trigger = {
		wants_world_news = yes
	}

	option = {
		name = flavornews.2.a
		trigger = {
			NOT = {
				tag = AUS
				tag = HUN
				tag = CZE
				tag = CRO
				tag = GAL
			}
		}
	}
	option = {
		name = flavornews.2.b
		trigger = {
			OR = {
				tag = AUS
				tag = HUN
				tag = CZE
				tag = CRO
				tag = GAL
			}
		}
		if = {
			limit = {
				tag = AUS
			}
			ALL_subtract_one_collapse = yes
		}
	}
}
