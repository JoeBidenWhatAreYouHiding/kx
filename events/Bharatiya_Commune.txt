				}
			}
		}
	}

	option = {
		name = beng.6.a

		set_politics = {
			ruling_party = totalist
			elections_allowed = no
		}
		remove_all_ministers = yes

		add_ideas = BHC_A_C_N_Nambiar_for_tot

			limit = {

				modifier = BHC_approves_government
