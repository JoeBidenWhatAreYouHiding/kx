country_event = {
	id = kr.hungary.301
	immediate = {log = "[GetDateText]: [Root.GetName]: event kr.hungary.301"}
	title = kr.hungary.301.t
	desc = kr.hungary.301.d
	picture = GFX_report_event_generic_diplomacy2

	is_triggered_only = yes

	option = {
		name = kr.hungary.301.a # We are triumphant!
		ai_chance = {
			factor = 100
		}
		add_political_power = 200
		if = {	
			limit = { CRO = { is_in_faction_with = AUS } }
			annex_country = {
				target = CRO
				transfer_troops = no
			}
		}

		AUS = { white_peace_with_ROOT = yes }
		

		# separate peace for Poland if it joined the war
		if = {
			limit = {
				POL = {
					has_country_flag = POL_joined_HUN_war
					has_war_with = AUS
				}
			}
			AUS = { white_peace = POL }
			GAL = {
				every_unit_leader = {
					set_unit_leader_flag = POL_leader
					set_nationality = POL
				}
			}
			POL = {
				annex_country = {
					target = GAL
					transfer_troops = no
				}
			}
