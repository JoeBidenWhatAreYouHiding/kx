country_event = { #Calculate necessary pressure
	id = bdeeubts.2
	immediate = { log = "[GetDateText]: [Root.GetName]: event bdeeubts.2" }
	title = bdeeubts.2.t
    desc = bdeeubts.2.d

	picture = GFX_event_AUS_panyuro_congress

	is_triggered_only = yes
	
	hidden = yes
	
	option = { #FREUDE SCHONER GOTTERFUNKEN TOCHTER AUS ELYSIUM
		name = bdeeubts.2.a
		if = {
			limit = {
				any_country = {
					EUU_EU_member = no
					capital_scope = {
						is_on_continent = europe
					}
				}
			}
			every_country = {
				limit = {
					EUU_EU_member = no
					capital_scope = {
						is_on_continent = europe
					}
				}
				EUU_calculate_necessary_pressure = yes
			}
			country_event = {
				id = bdeeubts.2
				days = 30
			}
		}
		if = {
			limit = {
				any_country = {
					EUU_EU_member = yes
					exists = no
				}
			}
			PREV = { EUU_remove_from_EU = yes }
		}
	}
}

news_event = { #Europe Unites
	id = bdeeunews.1
	immediate = { log = "[GetDateText]: [Root.GetName]: event bdeeunews.1" }
	title = bdeeunews.1.t
	desc = bdeeunews.1.desc
	
	picture = GFX_news_event_AUS_European_Unification
	
	major = yes

	show_major = {
		regional_europe_news = yes
	}
	
	trigger = {
		wants_world_news = yes
	}

	is_triggered_only = yes

	option = { #Europa awakens!
		name = bdeeunews.1.a
		trigger = { tag = EUU }
	}
	
	option = { #Who ever would've thought?
		name = bdeeunews.1.b
		trigger = {
			OR = {
				capital_scope = { NOT = { is_on_continent = europe } }
				is_in_faction_with = EUU
			}
			NOT = { tag = EUU }
		}
	}
	
	option = { #We must be careful.
		name = bdeeunews.1.c
		trigger = {
			NOT = {
				is_in_faction_with = EUU
				tag = EUU
			}
			is_major = yes
		}
	}

	option = { #That's suprising.
		name = bdeeunews.1.d
		trigger = {
			NOT = {
				capital_scope = { NOT = { is_on_continent = europe } }
				is_in_faction_with = EUU
				tag = EUU
			}
			is_major = no
		}
	}
}
