##Event: The Nationalist Union Party is Formed
country_event = {
	id = natpop.1
	title = natpop.1.title
	desc = natpop.1.desc
	fire_only_once = yes
	trigger = {
		has_country_flag= CHL_Chairman_Elections
	}
	mean_time_to_happen = {
		days = 4
	}
	option = {
		name = natpop.1.a
		ai_chance = { factor = 8}
		set_country_flag= CHL_Crackdown_Syndie
	}
	option = {
		name = natpop.1.b
		ai_chance = { factor = 4}
		set_country_flag= CHL_NatPop_Crackdown
		set_party_name = 
			{ 
		ideology = national_populist 
		long_name = CHL_national_populist_herrera 
		name = CHL_national_populist_herrera_short 
			}
	}
}

##Event: The Army Coups!
country_event = {
	id = natpop.5
	title = natpop.5.title
	desc = natpop.5.desc
	picture = GFX_event_test
	fire_only_once = yes
	trigger = {
		has_country_flag = Natpop_Army_Refused
	}
	mean_time_to_happen = {
		days = 5
	}
	option = {
		name = natpop.5.a
		ai_chance = { factor = 1}
		if = {
			limit = {
				has_government = national_populist
			}
			retire_country_leader = yes
		}
		set_politics = {
			ruling_party = national_populist 
			elections_allowed = no 
		}
		create_country_leader = {
			name = "Ariosto Herrera" 
			desc = " " 
			picture = "Portrait_CHL_Ariosto_Herrera.png" 
			expire = "1965.1.1"
			ideology = national_populist_subtype
			traits = { }
		}
	}
}
