	id = ins.1
	title = ins.1.t
	desc = ins.1.d
	picture = GFX_report_event_INS_independence_rally
	
	fire_only_once = yes

	trigger = {
		tag = INS
		OR = {
			focus_progress = { focus = INS_political_crossroad progress > 0.5 }
			has_completed_focus = INS_political_crossroad
		}
		has_government = authoritarian_democrat
		OR = {
			NOT = { has_country_flag = ins_high_unity }
			has_game_rule = {
				rule = INS_future
				option = INS_FUTURE_PATH_ISLAMIST
			}
		}
	}

	immediate = {
		log = "[GetDateText]: [Root.GetName]: event ins.1"
		hidden_effect = {
			country_event = {
				id = ins.2
				days = 20
			}
		}
	}
	
	mean_time_to_happen = {
		days = 1
	}

	option = { #Sukarno dies
		name = ins.1.b
		ai_chance = {
			factor = 80
		}
		add_stability = -0.1
		create_country_leader = {
			name = "Mohammed Hatta"
			picture = "gfx/leaders/INS/Portrait_Indonesia_Mohammad_Hatta.png"
			expire = "1.1.1"
			ideology = authoritarian_democrat_subtype
		}
		add_ideas = INS_Mohammed_Hatta_hog_ade
		set_country_flag = Sukarno_dead
	}

	option = { #Sukarno survives
		name = ins.1.a
		ai_chance = {
			factor = 20
		}
		add_stability = -0.1
	}
