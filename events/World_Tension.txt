add_namespace = worldtension

# Second Weltkrieg
country_event = {
	id = worldtension.1
	immediate = { log = "[GetDateText]: [Root.GetName]: event worldtension.1" }

	trigger = {
		GER = { has_war_with = FRA }
	}

	is_triggered_only = yes
	fire_only_once = yes
	hidden = yes

	option = {
		clr_global_flag = KR_italy_balance
		FRA = {
			add_named_threat = {
				threat = 50
				name = second_weltkrieg_threat
			}
		}
	}
}

# Second American Civil War
country_event = {
	id = worldtension.2
	immediate = { log = "[GetDateText]: [Root.GetName]: event worldtension.2" }

	trigger = {
		USA = {
			OR = {
				has_war_with = CSA
				has_war_with = TEX
			}
		}
	}

	is_triggered_only = yes
	fire_only_once = yes
	hidden = yes

	option = {
		if = {
			limit = {
				country_exists = CSA
				country_exists = TEX
			}
			USA = {
				add_named_threat = {
					threat = 15
					name = second_american_civil_war_threat
				}
			}
		}
		else = {
			USA = {
				add_named_threat = {
					threat = 10
					name = second_american_civil_war_threat
				}
			}
		}
	}
}
